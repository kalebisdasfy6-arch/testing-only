local function normalizeDropdownValue(value)
    if type(value) == "table" then
        if value[1] then return value[1] end
        for k, enabled in pairs(value) do if enabled then return k end end
        return nil
    end
    return value
end

local function setLibraryLabel(label, text)
    if not label then return end
    pcall(function()
        if type(label.Set) == "function" then label:Set(text); return end
        if type(label.SetText) == "function" then label:SetText(text); return end
        if label.Instance and label.Instance.Text ~= nil then label.Instance.Text = text; return end
        if label.Elements and label.Elements["Text"] and label.Elements["Text"].Instance then
            label.Elements["Text"].Instance.Text = text; return
        end
        if label.Text ~= nil then label.Text = text end
    end)
end

local function formatMoney(amount)
    local formatted = tostring(math.floor(tonumber(amount) or 0))
    while true do
        local newFormatted, count = formatted:gsub("^(-?%d+)(%d%d%d)", "%1,%2")
        formatted = newFormatted
        if count == 0 then break end
    end
    return "$" .. formatted
end

local CarFly = { Enabled = false, Speed = 150 }
local carFlyConnection = nil

local function StopCarFly()
    CarFly.Enabled = false
    if carFlyConnection then carFlyConnection:Disconnect(); carFlyConnection = nil end
    for _, v in pairs(workspace:GetDescendants()) do
        if v.Name == "SleepyHubCarVelocity" or v.Name == "SleepyHubCarGyro" then v:Destroy() end
    end
end

local function StartCarFly()
    local char = LocalPlayer.Character
    if not char then return end
    local hum = char:FindFirstChildWhichIsA("Humanoid")
    if not hum then return end
    local seat = hum.SeatPart
    if not seat or not seat:IsA("VehicleSeat") then
        sendNotification("Car Fly","Sit in a vehicle first!"); return
    end

    local root = (seat.Parent and seat.Parent.PrimaryPart) or seat
    StopCarFly()
    CarFly.Enabled = true

    local bv = Instance.new("BodyVelocity")
    bv.Name = "SleepyHubCarVelocity"
    bv.MaxForce = Vector3.new(1e9,1e9,1e9)
    bv.Parent = root

    local bg = Instance.new("BodyGyro")
    bg.Name = "SleepyHubCarGyro"
    bg.MaxTorque = Vector3.new(1e9,1e9,1e9)
    bg.P = 50000; bg.D = 1000
    bg.CFrame = root.CFrame
    bg.Parent = root

    carFlyConnection = RunService.RenderStepped:Connect(function()
        if not CarFly.Enabled or not seat.Parent then StopCarFly() return end
        local cam = workspace.CurrentCamera
        local moveDir = Vector3.zero
        local look = cam.CFrame.LookVector
        local right = cam.CFrame.RightVector
        if UserInputService:IsKeyDown(Enum.KeyCode.W) then moveDir = moveDir + look end
        if UserInputService:IsKeyDown(Enum.KeyCode.S) then moveDir = moveDir - look end
        if UserInputService:IsKeyDown(Enum.KeyCode.A) then moveDir = moveDir - right end
        if UserInputService:IsKeyDown(Enum.KeyCode.D) then moveDir = moveDir + right end
        if UserInputService:IsKeyDown(Enum.KeyCode.Space) then moveDir = moveDir + Vector3.new(0,1,0) end
        if UserInputService:IsKeyDown(Enum.KeyCode.LeftShift) then moveDir = moveDir - Vector3.new(0,1,0) end
        if moveDir.Magnitude > 0 then
            bv.Velocity = moveDir.Unit * CarFly.Speed
            bg.CFrame  = CFrame.new(root.Position, root.Position + moveDir)
        else
            bv.Velocity = Vector3.zero
        end
    end)
end

local function GetNearestCar(hrp)
    local CivCars = workspace:FindFirstChild("CivCars")
    if not CivCars then return nil end
    local nearestCar, nearestDist = nil, math.huge
    for _, car in ipairs(CivCars:GetChildren()) do
        if car:IsA("Model") then
            local seat = car:FindFirstChild("DriveSeat") or car:FindFirstChildWhichIsA("VehicleSeat")
            if seat and not seat.Occupant then
                if not car.PrimaryPart then
                    pcall(function() car.PrimaryPart = car:FindFirstChildWhichIsA("BasePart") end)
                end
                if car.PrimaryPart then
                    local d = (car.PrimaryPart.Position - hrp.Position).Magnitude
                    if d < nearestDist then nearestDist, nearestCar = d, car end
                end
            end
        end
    end
    return nearestCar
end

local function isValidTargetPlayer(p)
    if not p or p == LocalPlayer then return false end
    local c = p.Character; if not c then return false end
    local h = c:FindFirstChildOfClass("Humanoid")
    local r = c:FindFirstChild("HumanoidRootPart")
    return h and h.Health > 0 and r ~= nil
end

local function getNearestTargetPlayer()
    local lr = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")

    for _, p in ipairs(Players:GetPlayers()) do

            local d = (p.Character.HumanoidRootPart.Position - lr.Position).Magnitude
            if d < nd then n, nd = p, d end
        end
    end
    return n
end

local WeaponMods = {
    InfiniteAmmo=false, InfiniteClips=false, Bullets80k=false,
    ModifyRecoilValue=false, ModifySpreadValue=false, DisableJamming=false,
    InstantReload=false, InstantEquip=false, ModifyFireRate=false,
    Automatic=false, DamageAmplified=false, SpoofAccuracy=false,
    InfiniteDamage=false,
    ReloadSpeed=0.2, EquipSpeed=0.2,
    FireRateSpeed=50, SpreadPercentage=50, RecoilPercentage=50,
}
local MianiWeaponOriginals = {}
local GunOriginalColors = {}
local GunChams = false
local RainbowGun = false
local GunChamsColor = Color3.fromRGB(0,200,0)

local function getEquippedGunSetting()
    local character = LocalPlayer.Character
    if not character then return nil, nil end
    local tool = character:FindFirstChildOfClass("Tool")
    if not tool or not tool:FindFirstChild("Setting") then return nil, nil end
    local ok, settings = pcall(require, tool.Setting)
    if ok and type(settings) == "table" then return tool, settings end
end

local function applyAllWeaponMods()
    local tool, settings = getEquippedGunSetting()
    if not settings then return end
    pcall(function()
        if WeaponMods.InfiniteAmmo then settings.Ammo=99999; settings.AmmoPerMag=99999; settings.LimitedAmmoEnabled=false end
        if WeaponMods.InfiniteClips then settings.Ammo=80000; settings.AmmoPerMag=80000; settings.LimitedAmmoEnabled=false end
        if WeaponMods.Bullets80k then
            settings.Ammo=80000; settings.AmmoPerMag=80000; settings.LimitedAmmoEnabled=false
            local gunLocal = tool:FindFirstChild("GunScript_Local")
            if gunLocal and type(getsenv) == "function" then
                pcall(function()
                    local env = getsenv(gunLocal)
                    if env and type(env.Reload) == "function" and type(debug.setupvalue) == "function" then
                        debug.setupvalue(env.Reload, 1, 80000); debug.setupvalue(env.Reload, 3, 80000)
                    end
                end)
            end
        end
        if WeaponMods.ModifyRecoilValue then settings.Recoil=0; settings.CameraRecoilingEnabled=false end
        if WeaponMods.ModifySpreadValue then settings.Spread=0; settings.SpreadX=0; settings.SpreadY=0; settings.Accuracy=1 end
        if WeaponMods.DisableJamming then settings.JamChance=0 end
        if WeaponMods.InstantReload then settings.ReloadTime = WeaponMods.ReloadSpeed or 0.1 end
        if WeaponMods.InstantEquip then settings.EquipTime = WeaponMods.EquipSpeed or 0.1; settings.EquippingTime = WeaponMods.EquipSpeed or 0.1 end
        if WeaponMods.ModifyFireRate then settings.FireRate = 0 end
        if WeaponMods.Automatic then settings.Auto = true end
        if (WeaponMods.DamageAmplified or WeaponMods.InfiniteDamage) then settings.BaseDamage = 1e9 end
        if WeaponMods.SpoofAccuracy then settings.Accuracy = 1 end
    end)
end

LocalPlayer.CharacterAdded:Connect(function(character)
    character.ChildAdded:Connect(function(c) if c:IsA("Tool") then task.defer(applyAllWeaponMods) end end)
end)
if LocalPlayer.Character then

end

local function MianiUpdateGunColor()
    if not GunChams then return end
    local char = LocalPlayer.Character
    local tool = char and char:FindFirstChildOfClass("Tool")
    if not tool then return end
    for _, v in pairs(tool:GetDescendants()) do
        if v:IsA("BasePart") then
            if not GunOriginalColors[v] then GunOriginalColors[v] = v.Color end
            if RainbowGun then v.Color = Color3.fromHSV((tick() % 5) / 5, 1, 1)
            else v.Color = GunChamsColor or Color3.fromRGB(0,200,0) end
            v.Material = Enum.Material.Neon
        end
    end
end
local function MianiRestoreGunColor()
    for part, color in pairs(GunOriginalColors) do
        if part and part.Parent then part.Color=color; part.Material=Enum.Material.Plastic end
    end
    GunOriginalColors = {}
end

local function MianiGetEmoteAnimator()
    local char = LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()
    local hum  = char and char:FindFirstChildOfClass("Humanoid")
    if not hum then return nil end
    local a = hum:FindFirstChildOfClass("Animator")
    if not a then a = Instance.new("Animator"); a.Parent = hum end
    return a

local function MianiStopAllTracks()
    local a = MianiGetEmoteAnimator()
    if a then for _, t in ipairs(a:GetPlayingAnimationTracks()) do pcall(function() t:Stop(0.1) end) end end
    MianiCurrentEmoteTrack = nil
end
local function MianiPlayEmoteById(id, name)
    id = tonumber(id); if not id then return end
    local animator = MianiGetEmoteAnimator(); if not animator then return end
    MianiStopAllTracks()
    local anim = MianiEmoteAnimCache[id]
    if not anim then
        anim = Instance.new("Animation"); anim.AnimationId = "rbxassetid://" .. tostring(id)
        MianiEmoteAnimCache[id] = anim
    end
    local ok, track = pcall(function()
        local t = animator:LoadAnimation(anim)
        t.Priority = Enum.AnimationPriority.Action
        t.Looped = MianiEmoteSettings.Loop
        t:Play(0.1)
        return t
    end)
    if ok and track then
        MianiCurrentEmoteTrack = track
        sendNotification("Emotes","Playing: "..tostring(name or id))
    end
end

-- prompts & background loops
-- (Instant Interaction — using the 64th.lol approach: simple set/restore to defaults,
--  no original-value tracking, no MaxActivationDistance changes)
local promptConnections  = {}
local promptAddedConnection
local function setupPrompt(p, enableInstant)
    if not p or not p:IsA("ProximityPrompt") then return end
    if enableInstant then
        p.HoldDuration        = 0
        p.RequiresLineOfSight = false
        promptConnections[p]  = true
    else
        p.HoldDuration        = 1
        p.RequiresLineOfSight = true
        promptConnections[p]  = nil
    end
end
local function refreshPrompts(enableInstant)
    -- clear any leftover connection state
    for prompt, conn in pairs(promptConnections) do
        if typeof(conn) == "RBXScriptConnection" then conn:Disconnect() end
        promptConnections[prompt] = nil
    end

    -- apply to every existing prompt
    for _, p in ipairs(workspace:GetDescendants()) do
        if p:IsA("ProximityPrompt") then setupPrompt(p, enableInstant) end
    end

    -- watch for new prompts only while enabled
    if promptAddedConnection then promptAddedConnection:Disconnect(); promptAddedConnection = nil end
    if enableInstant then
        promptAddedConnection = workspace.DescendantAdded:Connect(function(d)
            if d:IsA("ProximityPrompt") then setupPrompt(d, true) end
        end)
    end
end

-- Money value helpers
local function valueToNumber(o) if not o then return nil end local ok,v=pcall(function() return o.Value end); if ok and tonumber(v) then return tonumber(v) end end
local function findMoneyValue(c, names, recursive)
    if not c then return nil end
    local lk={}; for _, n in ipairs(names) do lk[string.lower(n)] = true end
    for _, ch in ipairs(c:GetChildren()) do if lk[string.lower(ch.Name)] then local v=valueToNumber(ch); if v~=nil then return v end end end
    if recursive then for _, ch in ipairs(c:GetDescendants()) do if lk[string.lower(ch.Name)] then local v=valueToNumber(ch); if v~=nil then return v end end end end
end
local function getMyMoneyValues()
    local cN={"Money","Cash","Wallet","Bank","Coins"}
    local dN={"FilthyStack","FilthyMoney","DirtyMoney","DirtyCash","IllegalMoney","Dirty","Filthy"}
    local stored = LocalPlayer:FindFirstChild("stored") or LocalPlayer:FindFirstChild("Stored")
    local leader = LocalPlayer:FindFirstChild("leaderstats") or LocalPlayer:FindFirstChild("Leaderstats") or LocalPlayer:FindFirstChild("stats")
    local clean = findMoneyValue(stored,{"Money"},false) or findMoneyValue(leader,cN,false) or findMoneyValue(LocalPlayer,cN,false) or findMoneyValue(LocalPlayer,cN,true)
    local dirty = findMoneyValue(stored,{"FilthyStack"},false) or findMoneyValue(stored,dN,false) or findMoneyValue(LocalPlayer,dN,false) or findMoneyValue(LocalPlayer,dN,true)
    return clean, dirty
end

-- ESP drawing
local function createDrawing(className, props)
    if not Drawing or type(Drawing.new) ~= "function" then return {Visible=false, Remove=function() end} end
    local ok, obj = pcall(Drawing.new, className)
    if not ok or not obj then return {Visible=false, Remove=function() end} end
    for k, v in pairs(props or {}) do pcall(function() obj[k] = v end) end
    return obj
end

local function createESPObject(player)
    if player == LocalPlayer or ESPObjects[player] then return end
    local box = {}
    for _, n in ipairs({"Top","Bottom","Left","Right","TopLeft","TopRight","BottomLeft","BottomRight"}) do
        box[n] = createDrawing("Line",{Visible=false, Thickness=1, Color=ESPFlags.BoxColor})
    end
    local skeleton = {}
    for i = 1, #SkeletonPairs do skeleton[i] = createDrawing("Line",{Visible=false, Thickness=1, Color=ESPFlags.SkeletonColor}) end
    ESPObjects[player] = {
        Box=box,
        Fill          = createDrawing("Square",{Visible=false, Filled=true, Transparency=0.85, Color=ESPFlags.BoxFillColor}),
        HealthOutline = createDrawing("Square",{Visible=false, Filled=false, Color=Color3.fromRGB(0,0,0), Thickness=1}),
        HealthFill    = createDrawing("Square",{Visible=false, Filled=true, Color=Color3.fromRGB(0,255,0)}),
        HealthText    = createDrawing("Text",{Visible=false, Center=true, Outline=true, Size=ESPFlags.TextSize, Color=Color3.fromRGB(255,255,255)}),
        Name          = createDrawing("Text",{Visible=false, Center=true, Outline=true, Size=ESPFlags.TextSize, Color=ESPFlags.TextColor}),
        Distance      = createDrawing("Text",{Visible=false, Center=true, Outline=true, Size=ESPFlags.TextSize, Color=ESPFlags.TextColor}),
        Weapon        = createDrawing("Text",{Visible=false, Center=true, Outline=true, Size=ESPFlags.TextSize, Color=ESPFlags.TextColor}),
        Snapline      = createDrawing("Line",{Visible=false, Thickness=1, Color=ESPFlags.SnaplineColor}),
        LookTracer    = createDrawing("Line",{Visible=false, Thickness=1, Color=ESPFlags.LookTracerColor}),
        Skeleton      = skeleton,
    }
end
local function hideESPObject(obj)
    if not obj then return end
    if obj.Box then for _, l in pairs(obj.Box) do l.Visible=false end end
    if obj.Skeleton then for _, l in ipairs(obj.Skeleton) do l.Visible=false end end
    for _, k in ipairs({"Fill","HealthOutline","HealthFill","HealthText","Name","Distance","Weapon","Snapline","LookTracer"}) do
        if obj[k] then obj[k].Visible=false end
    end
end
local function removeESPObject(player)
    local obj = ESPObjects[player]
    if obj then
        if obj.Box then for _, l in pairs(obj.Box) do pcall(function() l:Remove() end) end end
        if obj.Skeleton then for _, l in ipairs(obj.Skeleton) do pcall(function() l:Remove() end) end end
        for _, k in ipairs({"Fill","HealthOutline","HealthFill","HealthText","Name","Distance","Weapon","Snapline","LookTracer"}) do
            if obj[k] then pcall(function() obj[k]:Remove() end) end
        end
        ESPObjects[player] = nil
    end
    if HighlightObjects[player] then pcall(function() HighlightObjects[player]:Destroy() end); HighlightObjects[player]=nil end
end
local function getESPColor(player, def)
    if ESPFlags.TeamColor and player.Team and LocalPlayer.Team and player.Team == LocalPlayer.Team then
        return Color3.fromRGB(0,120,255)
    end
    return def
end

end

local _espHiddenOnce = false
RunService.RenderStepped:Connect(function()
    -- Fast path: ESP off → hide everything ONCE then do nothing per-frame
    if not ESPFlags.Enabled then
        if not _espHiddenOnce then
            _espHiddenOnce = true
            for _, p in ipairs(Players:GetPlayers()) do
                hideESPObject(ESPObjects[p])
                if HighlightObjects[p] then HighlightObjects[p].Enabled = false end
            end
        end
        return
    end
    _espHiddenOnce = false
    local cam = workspace.CurrentCamera; if not cam then return end
    for _, p in ipairs(Players:GetPlayers()) do
        if p == LocalPlayer then continue end
        createESPObject(p)
        local obj = ESPObjects[p]
        local char = p.Character
        local hum  = char and char:FindFirstChildOfClass("Humanoid")
        local root = char and char:FindFirstChild("HumanoidRootPart")
        if not (obj and char and hum and root and hum.Health > 0) then
            hideESPObject(obj)
            if HighlightObjects[p] then HighlightObjects[p].Enabled=false end
            continue
        end
        local localRoot = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
        local distance = localRoot and (root.Position - localRoot.Position).Magnitude or (root.Position - cam.CFrame.Position).Magnitude
        if distance > ESPFlags.RenderDistance then
            hideESPObject(obj); if HighlightObjects[p] then HighlightObjects[p].Enabled=false end; continue
        end
        local rootPos, onScreen = cam:WorldToViewportPoint(root.Position)
        if not onScreen then hideESPObject(obj); continue end

        local size = char:GetExtentsSize()
        local top    = cam:WorldToViewportPoint((root.CFrame * CFrame.new(0, size.Y/2, 0)).Position)
        local bottom = cam:WorldToViewportPoint((root.CFrame * CFrame.new(0,-size.Y/2, 0)).Position)
        local height = math.max(math.abs(bottom.Y - top.Y), 2)
        local width  = math.max(height * 0.58, 2)
        local x, y   = top.X - width/2, top.Y
        local boxColor = getESPColor(p, ESPFlags.BoxColor)
        local textColor = ESPFlags.TextColor

        if ESPFlags.BoxFill then
            obj.Fill.Position=Vector2.new(x,y); obj.Fill.Size=Vector2.new(width,height)
            obj.Fill.Color=ESPFlags.BoxFillColor; obj.Fill.Transparency=0.85; obj.Fill.Visible=true
        else obj.Fill.Visible=false end

        if ESPFlags.Boxes then
            local tl,tr = Vector2.new(x,y), Vector2.new(x+width,y)
            local bl,br = Vector2.new(x,y+height), Vector2.new(x+width,y+height)
            local corner = math.min(width,height)*0.28
            if ESPFlags.BoxType=="Full" then
                updateLine(obj.Box.Top,tl,tr,boxColor); updateLine(obj.Box.Bottom,bl,br,boxColor)

        ["Fists"]=true, ["Fist"]=true, ["Phone"]=true, ["Car Keys"]=true,
        ["Car keys"]=true, ["Bandage"]=true, ["762s"]=true, ["556s"]=true,

        local hds = workspace:FindFirstChild("HomeDoorSS")
        local hd  = hds and hds:FindFirstChild("HomeDoor")
        return hd and hd:FindFirstChild("Safe")
    end

    local function GetSafeItems()
        local items = {}
        local invData = LocalPlayer:FindFirstChild("InvData")
        if invData then for _, c in ipairs(invData:GetChildren()) do table.insert(items, c.Name) end end
        table.sort(items); return items
    end

    local function GetAllBackpackItems()
        local items = {}
        local char = LocalPlayer.Character
        local bp   = LocalPlayer:FindFirstChild("Backpack")
        if char then for _, t in pairs(char:GetChildren()) do
            if t:IsA("Tool") and not dupeBlacklist[t.Name] and not table.find(items, t.Name) then table.insert(items, t.Name) end
        end end
        if bp then for _, t in pairs(bp:GetChildren()) do
            if t:IsA("Tool") and not dupeBlacklist[t.Name] and not table.find(items, t.Name) then table.insert(items, t.Name) end
        end end
        table.sort(items); return items
    end

    local function GetLockedTools()
        local tools = {}
        if LocalPlayer:FindFirstChild("Backpack") then
            for _, t in pairs(LocalPlayer.Backpack:GetChildren()) do
                if t:IsA("Tool") and not dupeBlacklist[t.Name] then table.insert(tools, t.Name) end
            end
        end
        return tools
    end

    local function FirePromptsInArea(range, nameFilter)
        local char = LocalPlayer.Character
        local root = char and char:FindFirstChild("HumanoidRootPart")
        if not root then return end
        for _, v in pairs(workspace:GetDescendants()) do
            if v:IsA("ProximityPrompt") then
                local pPos = (v.Parent:IsA("Model") and v.Parent:GetPivot().Position)
                          or (v.Parent:IsA("BasePart") and v.Parent.Position)
                if pPos and (pPos - root.Position).Magnitude < range then
                    if not nameFilter
                    or (v.Parent.Name:lower():find(nameFilter:lower()) or v.Name:lower():find(nameFilter:lower())) then
                        pcall(function() fireproximityprompt(v) end)
                    end
                end
            end
        end
    end

    -- ============== TAKE / STORE ==============
    local function TakeFromSafe(itemName)
        local SafeFolder = getSafeFolder()
        if not SafeFolder then sendNotification("Safe","Safe not found in this server."); return false end
        if not InventoryRemote then sendNotification("Safe","InventoryRemote missing."); return false end
        local safePart = SafeFolder:FindFirstChild("Union") or SafeFolder:FindFirstChildWhichIsA("BasePart")
        if not safePart then return false end
        local char = LocalPlayer.Character
        local root = char and char:FindFirstChild("HumanoidRootPart")
        if not root then return false end
        local originalCF = root.CFrame
        pcall(function()
            Config:Teleport(safePart.CFrame * CFrame.new(0, 2, 0))
            task.wait(0.3)
            InventoryRemote:FireServer("Change", itemName, "Inv", SafeFolder)
            task.wait(0.5)
            Config:Teleport(originalCF)
            task.wait(0.5)
        end)
        local got = LocalPlayer.Backpack:FindFirstChild(itemName)
                  or (LocalPlayer.Character and LocalPlayer.Character:FindFirstChild(itemName))
        sendNotification("Safe", got and ("Took " .. itemName) or ("Failed: " .. itemName))
        return got ~= nil
    end

    local function StoreHoldingItem()
        local char = LocalPlayer.Character; if not char then return end
        local tool = char:FindFirstChildOfClass("Tool")
        if not tool then sendNotification("Safe","Hold an item first!"); return end
        local SafeFolder = getSafeFolder()
        if not SafeFolder then sendNotification("Safe","Safe not found."); return end
        local safePart = SafeFolder:FindFirstChild("Union") or SafeFolder:FindFirstChildWhichIsA("BasePart")
        local root = char:FindFirstChild("HumanoidRootPart")
        if not safePart or not root then return end
        local originalCF = root.CFrame
        local hum = char:FindFirstChildOfClass("Humanoid")
        if hum then hum:UnequipTools() end
        task.wait(0.2)
        Config:Teleport(safePart.CFrame * CFrame.new(0, 2, 0))
        task.wait(0.3)
        pcall(function()
            if BackpackRemoteSafe then BackpackRemoteSafe:InvokeServer("Store", tool.Name) end
            if InventoryRemote then InventoryRemote:FireServer("Change", tool.Name, "Backpack", SafeFolder) end
        end)
        task.wait(0.5)
        Config:Teleport(originalCF)
        sendNotification("Safe","Stored " .. tool.Name)
    end

    -- ============== SAFE DUPE ==============
    local function SafeDupeOnce(itemName, returnCFrame)
        local SafeFolder = getSafeFolder()
        if not SafeFolder then return false end
        local safePart = SafeFolder:FindFirstChild("Union") or SafeFolder:FindFirstChildWhichIsA("BasePart")
        if not safePart then return false end
        pcall(function()
            Config:Teleport(safePart.CFrame * CFrame.new(0, 2, 0))
            task.wait(0.3)
            task.spawn(function() if BackpackRemoteSafe then BackpackRemoteSafe:InvokeServer("Store", itemName) end end)
            task.spawn(function() if InventoryRemote then InventoryRemote:FireServer("Change", itemName, "Backpack", SafeFolder) end end)
            task.wait(0.6)
            Config:Teleport(returnCFrame)
            task.wait(1.2)
            if BackpackRemoteSafe then BackpackRemoteSafe:InvokeServer("Grab", itemName) end
            task.wait(0.5)
        end)
        return true
    end

    local autoDupeActive = false
    local dupeSpeed      = 15
    local dupeCounter    = 0
    local selectedDupeItem = nil   -- nil = dupe ALL locked tools

    local function StartSafeDupe(statusLabelFn)
        local char = LocalPlayer.Character
        local root = char and char:FindFirstChild("HumanoidRootPart")
        if not root then sendNotification("Dupe","No character!"); return end
        local origLoc = root.CFrame

        autoDupeActive = true
        sendNotification("Dupe","Starting safe dupe...")

        task.spawn(function()
            -- "Rent" + "Open Safe" sequence (Da Hood positions)
            statusLabelFn("RENTING...")
            Config:Teleport(CFrame.new(-1443, 256, 2137)); task.wait(0.7); FirePromptsInArea(15)
            statusLabelFn("OPENING SAFE...")
            Config:Teleport(CFrame.new(-1473, 256, 2134)); task.wait(0.7); FirePromptsInArea(15, "Safe")
            statusLabelFn("RETURNING...")
            Config:Teleport(origLoc); task.wait(0.5)

            local targets
            if selectedDupeItem then
                targets = { selectedDupeItem }
                -- equip first
                pcall(function()
                    if BackpackRemoteSafe then BackpackRemoteSafe:InvokeServer("Grab", selectedDupeItem) end
                    task.wait(0.5)
                    local hum = char and char:FindFirstChildOfClass("Humanoid")
                    if hum then hum:UnequipTools() end
                end)
            else
                targets = GetLockedTools()
                if #targets == 0 then
                    statusLabelFn("NO TOOLS")
                    autoDupeActive = false
                    sendNotification("Dupe","No tools in backpack.")
                    return
                end
            end

            for i = 1, dupeSpeed do
                if not autoDupeActive then break end
                local item = targets[((i - 1) % #targets) + 1]
                statusLabelFn(string.format("DUPE %d/%d: %s", i, dupeSpeed, tostring(item):upper()))
                SafeDupeOnce(item, origLoc)
                dupeCounter = dupeCounter + 1
                task.wait(0.2)
            end

            autoDupeActive = false
            statusLabelFn("DONE ✓  (total duped: " .. dupeCounter .. ")")
            sendNotification("Dupe", dupeSpeed .. " cycles done.")
        end)
    end

    local function StopSafeDupe(statusLabelFn)
        autoDupeActive = false
        statusLabelFn("STOPPED")
        sendNotification("Dupe","Stopped.")
    end

    -- ============== AUTO DROP ==============
    local autoDropEnabled = false
    local function ToggleAutoDrop(state, statusLabelFn)
        autoDropEnabled = state
        sendNotification("Auto Drop", state and "Started" or "Stopped")
        if state then
            task.spawn(function()
                statusLabelFn("DROPPING...")
                while autoDropEnabled do
                    local bp = LocalPlayer:FindFirstChild("Backpack")
                    local ch = LocalPlayer.Character
                    if not bp or not ch then task.wait(0.5); continue end
                    local found = false
                    for _, item in ipairs(bp:GetChildren()) do
                        if not autoDropEnabled then break end
                        if item:IsA("Tool") and not dupeBlacklist[item.Name] then
                            found = true
                            local hum = ch:FindFirstChildOfClass("Humanoid")
                            if hum then
                                hum:EquipTool(item)
                                local dt = tick()
                                repeat
                                    if DropItemRemote then

            btn.BackgroundColor3 = ACCENT_DIM

        btn.MouseLeave:Connect(function()
            btn.BackgroundColor3 = origColor
        end)
    end

------------------------------------------------------------
--  FAST TAB SWITCH PATCH
--  The library's default Page:Turn loops through EVERY descendant
--  of the page and creates a TweenService tween for it. With many
--  sections that's hundreds of tweens per swap → big stutter.
--  Replace it with an instant Visible toggle.
------------------------------------------------------------
do
    local Theme = Library.Theme or {}
    local AccentColor = Theme.Accent or PURPLE
    local TextColor   = Theme.Text   or TEXT
    if Window and Window.Pages then
        for _, page in ipairs(Window.Pages) do
            page.Turn = function(self, Bool)
                if self.Active == Bool then return end
                self.Active = Bool
                local pageInst = self.Elements and self.Elements["Page"] and self.Elements["Page"].Instance
                local textInst = self.Elements and self.Elements["Text"] and self.Elements["Text"].Instance
                local hideInst = self.Elements and self.Elements["Hide"] and self.Elements["Hide"].Instance
                if pageInst then pageInst.Visible = Bool end
                if textInst then
                    textInst.TextColor3      = Bool and AccentColor or TextColor
                    textInst.TextTransparency = Bool and 0 or 0.5
                end
                if hideInst then hideInst.Visible = Bool end
            end
        end
    end
end

------------------------------------------------------------
--  UI SCALE  (bigger on mobile / iPad)
--  Hits every ScreenGui + every UIScale inside it, then re-applies
--  repeatedly so the library cannot revert it.
------------------------------------------------------------
local function applyUIScale()
    local UIS = game:GetService("UserInputService")
    local isTouchDevice = UIS.TouchEnabled
    local screen = workspace.CurrentCamera and workspace.CurrentCamera.ViewportSize or Vector2.new(1920, 1080)
    local isTablet = isTouchDevice and (screen.X >= 900 or screen.Y >= 700)
    local isMobile  = isTouchDevice and not isTablet

    local target
    if isTablet then
        target = 2.6
    elseif isMobile then
        target = 2.9
    else
        target = 1.15
    end

    local roots = { CoreGui, LocalPlayer:FindFirstChildOfClass("PlayerGui") }
    for _, root in ipairs(roots) do
        if not root then continue end
        for _, sg in ipairs(root:GetChildren()) do
            if not sg:IsA("ScreenGui") then continue end
            if sg.Name == "MiamiFootballToggleMenu" then continue end
            for _, d in ipairs(sg:GetDescendants()) do
                if d:IsA("UIScale") then
                    pcall(function() d.Scale = target end)
                end
            end
            local direct = sg:FindFirstChildOfClass("UIScale")
            if not direct then
                local s = Instance.new("UIScale")
                s.Scale = target
                s.Parent = sg
            else
                pcall(function() direct.Scale = target end)
            end
        end
    end
end

task.spawn(function()
    task.wait(2)
    applyUIScale()
    for _ = 1, 20 do
        task.wait(0.5)
        applyUIScale()
    end
end)
