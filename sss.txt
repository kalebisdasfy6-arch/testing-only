local function collectShopItems()
    local items = {}
    local promptPurchases = workspace:FindFirstChild("PromptPurchases")
    if promptPurchases then
        for _, value in pairs(promptPurchases:GetChildren()) do
            local price = value:FindFirstChild("Price")
            local proxprompt = value:FindFirstChild("proxprompt")
            local prompt = proxprompt and proxprompt:FindFirstChildOfClass("ProximityPrompt")
            if price and proxprompt and prompt then
                local gunWithPrice = string.format("%s - $%s", tostring(value), tostring(price.Value))
                table.insert(items, gunWithPrice)
            end
        end
    end
    local gunsFolder = workspace:FindFirstChild("GUNS")
    if gunsFolder then
        for _, value in pairs(gunsFolder:GetChildren()) do
            local price = value:FindFirstChild("Price", true)
            local prompt = value:FindFirstChildWhichIsA("ProximityPrompt", true)
            if price and prompt then
                local gunWithPrice = string.format("%s - $%s", tostring(value), tostring(price.Value))
                if not table.find(items, gunWithPrice) then
                    table.insert(items, gunWithPrice)
                end
            end
        end
    end
    if #items == 0 then
        items = {"Shiesty", "WhiteShiesty", "Water", "RawSteak", "RawChicken", "Bagelite", ".45 Infinity", "BluGloves", "WhiteGloves", "BlackGloves", "RedGloves"}
    end
    return items
end

local function DrawBulletTracer(EndPos)
    if Tracer_Delay then return end
    Tracer_Delay = true
    pcall(function()
        local tool = LocalPlayer.Character and LocalPlayer.Character:FindFirstChildOfClass("Tool")
        if not tool or not tool:FindFirstChild("Handle") then return end
        
        local muzzle = tool.Handle:FindFirstChild("GunMuzzlePoint") or tool.Handle:FindFirstChild("GunFirePoint") or tool.Handle
        local startPos = muzzle:IsA("Attachment") and muzzle.WorldCFrame.Position or muzzle.Position
        
        local StartPart = Instance.new("Part")
        StartPart.Anchored = true; StartPart.Transparency = 1; StartPart.Position = startPos
        StartPart.Parent = workspace; StartPart.CanCollide = false; StartPart.CanQuery = false; StartPart.CanTouch = false

        local EndPart = Instance.new("Part")
        EndPart.Anchored = true; EndPart.CanCollide = false; EndPart.Transparency = 1
        EndPart.Position = EndPos; EndPart.Parent = workspace; EndPart.CanQuery = false; EndPart.CanTouch = false

        local Attachment1 = Instance.new("Attachment", StartPart)
        local Attachment2 = Instance.new("Attachment", EndPart)
        local Beam = Instance.new("Beam", StartPart)
        Beam.Attachment0 = Attachment1; Beam.Attachment1 = Attachment2
        Beam.FaceCamera = true; Beam.LightEmission = 1; Beam.LightInfluence = 0
        Beam.Width0 = 0.05; Beam.Width1 = 0.05

        if Tracers.Rainbow then
            task.spawn(function()
                while Beam and Beam.Parent do
                    local Hue = tick() % 5 / 5
                    local NextHue = (Hue + 0.1) % 1
                    Beam.Color = ColorSequence.new({ ColorSequenceKeypoint.new(0, Color3.fromHSV(Hue, 1, 1)), ColorSequenceKeypoint.new(1, Color3.fromHSV(NextHue, 1, 1)) })
                    task.wait(0.05)
                end
            end)
        else
            Beam.Color = ColorSequence.new({ ColorSequenceKeypoint.new(0, Tracers.StartColor), ColorSequenceKeypoint.new(1, Tracers.EndColor) })
        end

        task.delay(Tracers.Duration, function()
            StartPart:Destroy()
            EndPart:Destroy()
        end)
    end)
    task.spawn(function() task.wait(0.05); Tracer_Delay = false end)
end

local loadedEmoteCache = {}
local currentEmoteTrack = nil
local function playCustomEmote(id)
    task.spawn(function()
        local character = LocalPlayer.Character
        local humanoid = character and character:FindFirstChildOfClass("Humanoid")
        local animator = humanoid and humanoid:FindFirstChildOfClass("Animator")
        if not animator then return end
        if currentEmoteTrack then pcall(function() currentEmoteTrack:Stop() end) end
        
        local anim = loadedEmoteCache[id]
        if not anim then
            local success, objects = pcall(function()
                return game:GetObjects("rbxassetid://" .. tostring(id))
            end)
            if success and objects and #objects > 0 then
                local item = objects[1]
                if item:IsA("Animation") then
                    anim = item
                else
                    anim = item:FindFirstChildWhichIsA("Animation", true)
                end
            end
            if not anim then
                anim = Instance.new("Animation")
                anim.AnimationId = "rbxassetid://" .. tostring(id)
            end
            loadedEmoteCache[id] = anim
        end
        
        pcall(function()
            currentEmoteTrack = animator:LoadAnimation(anim)
            currentEmoteTrack.Priority = Enum.AnimationPriority.Action4
            currentEmoteTrack.Looped = true
            currentEmoteTrack:Play()
        end)
        sendNotification("Emote", "Playing emote!")
    end)
end

local CarFly = { Enabled = false, Speed = 150 }
local carFlyConnection

local function StopCarFly()
    CarFly.Enabled = false
    if carFlyConnection then carFlyConnection:Disconnect(); carFlyConnection = nil end
    for _, v in pairs(workspace:GetDescendants()) do
        if v.Name == "SleepyHubCarVelocity" or v.Name == "SleepyHubCarGyro" then v:Destroy() end
    end
end

local function StartCarFly()
    local char = LocalPlayer.Character
    if not char then return end
    local humanoid = char:FindFirstChildWhichIsA("Humanoid")
    if not humanoid then return end
    local seat = humanoid.SeatPart
    if not seat or not seat:IsA("VehicleSeat") then
        sendNotification("Car Fly", "Sit in a vehicle first!")
        return
    end
    local root = seat.Parent.PrimaryPart or seat
    StopCarFly()
    CarFly.Enabled = true
    local bv = Instance.new("BodyVelocity")
    bv.Name = "SleepyHubCarVelocity"
    bv.MaxForce = Vector3.new(1e9,1e9,1e9)
    bv.Parent = root
    local bg = Instance.new("BodyGyro")
    bg.Name = "SleepyHubCarGyro"
    bg.MaxTorque = Vector3.new(1e9,1e9,1e9)
    bg.P = 50000
    bg.D = 1000
    bg.CFrame = root.CFrame
    bg.Parent = root
    carFlyConnection = RunService.RenderStepped:Connect(function()
        if not CarFly.Enabled then StopCarFly() return end
        if not seat.Parent then StopCarFly() return end
        local cam = workspace.CurrentCamera
        local moveDir = Vector3.zero
        local look = cam.CFrame.LookVector
        local right = cam.CFrame.RightVector
        if UserInputService:IsKeyDown(Enum.KeyCode.W) then moveDir = moveDir + look end
        if UserInputService:IsKeyDown(Enum.KeyCode.S) then moveDir = moveDir - look end
        if UserInputService:IsKeyDown(Enum.KeyCode.A) then moveDir = moveDir - right end
        if UserInputService:IsKeyDown(Enum.KeyCode.D) then moveDir = moveDir + right end
        if UserInputService:IsKeyDown(Enum.KeyCode.Space) then moveDir = moveDir + Vector3.new(0,1,0) end
        if UserInputService:IsKeyDown(Enum.KeyCode.LeftShift) then moveDir = moveDir - Vector3.new(0,1,0) end
        if moveDir.Magnitude > 0 then
            bv.Velocity = moveDir.Unit * CarFly.Speed
            bg.CFrame = CFrame.new(root.Position, root.Position + moveDir)
        else
            bv.Velocity = Vector3.zero
        end
    end)
end

local function GetNearestCar(hrp)
    local CivCars = workspace:FindFirstChild("CivCars")
    if not CivCars then return nil end
    local nearestCar = nil
    local nearestDist = math.huge
    for _, car in ipairs(CivCars:GetChildren()) do
        if not car:IsA("Model") then continue end
        local seat = car:FindFirstChild("DriveSeat") or car:FindFirstChildWhichIsA("VehicleSeat")
        if not seat or seat.Occupant then continue end
        if not car.PrimaryPart then
            car.PrimaryPart = car:FindFirstChildWhichIsA("BasePart")
        end
        if not car.PrimaryPart then continue end
        local dist = (car.PrimaryPart.Position - hrp.Position).Magnitude
        if dist < nearestDist then
            nearestDist = dist
            nearestCar = car
        end
    end
    return nearestCar
end

local function formatMoney(amount)
    local formatted = tostring(math.floor(tonumber(amount) or 0))
    while true do
        local newFormatted, count = formatted:gsub("^(-?%d+)(%d%d%d)", "%1,%2")
        formatted = newFormatted
        if count == 0 then break end
    end
    return "$" .. formatted
end

local function getPing()
    local ok, result = pcall(function()
        return tonumber(string.match(LocalPlayer:GetNetworkPing(), '%d+'))
    end)
    if ok and result then return result end
    return 100
end

local function wallCheckCharacter(character)
    if not character or not character:FindFirstChild("Head") then return false end
    local origin = Camera.CFrame.Position
    local params = RaycastParams.new()
    params.FilterDescendantsInstances = { LocalPlayer.Character, Camera, character }
    params.FilterType = Enum.RaycastFilterType.Blacklist
    params.IgnoreWater = true
    return workspace:Raycast(origin, character.Head.Position - origin, params) == nil
end

-- ========== DUPE / ITEM ==========
local function duplicateCurrentItem()
    if DupeCooldown then warn("Cooldown! Please wait."); return end
    DupeCooldown = true
    local Character = LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()
    local Backpack = LocalPlayer:WaitForChild('Backpack')
    local Tool = Character:FindFirstChildOfClass('Tool')
    if not Tool then
        warn("Couldn't find your tool! Please make sure you're holding one.")
        DupeCooldown = false
        return
    end
    Tool.Parent = Backpack
    task.wait(0.5)
    local ToolName = Tool.Name
    local ToolId = nil
    local ping = getPing()
    local delay = 0.25 + ((math.clamp(ping, 0, 300) / 300) * 0.03)

    local Connection = ReplicatedStorage.MarketItems.ChildAdded:Connect(function(item)
        if item.Name == ToolName then
            local owner = item:WaitForChild('owner', 2)
            if owner and owner.Value == LocalPlayer.Name then
                ToolId = item:GetAttribute('SpecialId')
            end
        end
    end)

    pcall(function() ReplicatedStorage.ListWeaponRemote:FireServer(ToolName, 676767) end)
    task.wait(delay)
    pcall(function() ReplicatedStorage.BackpackRemote:InvokeServer('Store', ToolName) end)
    task.wait(3)
    if ToolId then pcall(function() ReplicatedStorage.BuyItemRemote:FireServer(ToolName, 'Remove', ToolId) end) end
    pcall(function() ReplicatedStorage.BackpackRemote:InvokeServer('Grab', ToolName) end)
    task.wait(1)
    local dupedTool = Backpack:FindFirstChild(ToolName)
    if dupedTool then dupedTool.Parent = Character end
    Connection:Disconnect()
    task.wait(2)
    DupeCooldown = false
end

        if not AutoDupeCurrentItem then continue end
        if not DupeToolName or DupeToolName == "nil" then continue end
        if not DupeCooldown then duplicateCurrentItem() end
    end
end)

local function GetPlayerMoneyAmount()
    local pl = Players.LocalPlayer
    if not pl then return nil end
    local leader = pl:FindFirstChild("leaderstats") or pl:FindFirstChild("Leaderstats") or pl:FindFirstChild("stats")
    if leader then
        for _, name in ipairs({"Money","Cash","Bank","Coins","Wallet"}) do
            local val = leader:FindFirstChild(name)
            if val and type(val.Value) == "number" then return val.Value end
        end
    end
    if pl.GetAttribute then
        local a = pl:GetAttribute("Money") or pl:GetAttribute("Cash") or pl:GetAttribute("Coins")
        if a and tonumber(a) then return tonumber(a) end
    end

end

local function DropMoneyAmount(Amount)
    local BankRemote = ReplicatedStorage:FindFirstChild('BankProcessRemote', true)
    if not BankRemote then return false, "BankProcessRemote not found" end
    local function send(...)
        if BankRemote.InvokeServer then return BankRemote:InvokeServer(...) end
        if BankRemote.FireServer then return BankRemote:FireServer(...) end
        return nil
    end
    local attempts = {
        {"Drop", tostring(Amount)}, {"Drop", tonumber(Amount) or Amount},
        {{"Drop", tostring(Amount)}}, {{"Drop", tonumber(Amount) or Amount}},
    }
    for _, args in ipairs(attempts) do
        local ok, result = pcall(send, unpack(args))
        if ok and result ~= false then return true, result end
    end
    return false, "All drop attempts failed"
end

local function applyAllWeaponMods()
    local tool, settings = getEquippedGunSetting()
    if not settings then return end
    pcall(function()
        if WeaponMods.InfiniteAmmo then
            settings.Ammo = 99999; settings.AmmoPerMag = 99999; settings.LimitedAmmoEnabled = false
        end
        if WeaponMods.InfiniteClips then
            settings.Ammo = 80000; settings.AmmoPerMag = 80000; settings.LimitedAmmoEnabled = false
        end
        if WeaponMods.Bullets80k then
            settings.Ammo = 80000; settings.AmmoPerMag = 80000; settings.LimitedAmmoEnabled = false
            local gunLocal = tool:FindFirstChild("GunScript_Local")
            if gunLocal and type(getsenv) == "function" then
                pcall(function()
                    local env = getsenv(gunLocal)
                    if env and type(env.Reload) == "function" and type(debug.setupvalue) == "function" then
                        debug.setupvalue(env.Reload, 1, 80000)
                        debug.setupvalue(env.Reload, 3, 80000)
                    end
                end)
            end
        end
        if WeaponMods.ModifyRecoilValue then settings.Recoil = 0; settings.CameraRecoilingEnabled = false end
        if WeaponMods.ModifySpreadValue then settings.Spread = 0; settings.SpreadX = 0; settings.SpreadY = 0; settings.Accuracy = 1 end
        if WeaponMods.DisableJamming then settings.JamChance = 0 end
        if WeaponMods.InstantReload then settings.ReloadTime = WeaponMods.ReloadSpeed or 0.1 end
        if WeaponMods.InstantEquip then settings.EquipTime = WeaponMods.EquipSpeed or 0.1; settings.EquippingTime = WeaponMods.EquipSpeed or 0.1 end
        if WeaponMods.ModifyFireRate then settings.FireRate = 0 end
        if WeaponMods.Automatic then settings.Auto = true end
        if WeaponMods.DamageAmplified then settings.BaseDamage = 10000 end
        if WeaponMods.SpoofAccuracy then settings.Accuracy = 1 end
    end)
end

local function getNearestPlayer(maxDistance)
    local localRoot = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
    if not localRoot then return nil end
    local nearest, nearestDistance = nil, maxDistance or math.huge
    for _, player in ipairs(Players:GetPlayers()) do
        if isValidCombatPlayer(player) then
            
            local distance = (player.Character.HumanoidRootPart.Position - localRoot.Position).Magnitude
            if distance <= nearestDistance then
                nearestDistance = distance
                nearest = player
            end
        end
    end
    return nearest, nearestDistance
end

local function applyHitboxExpander()
    for _, player in ipairs(Players:GetPlayers()) do
        if player == LocalPlayer or not player.Character then continue end
        local partName = "Head"
        local part = player.Character:FindFirstChild(partName)
        if part and part:IsA("BasePart") then
            if WeaponMods.HitboxExpander then
                local size = WeaponMods.HitboxSize * 2
                part.Size = Vector3.new(size, size, size)
                part.Transparency = WeaponMods.HitboxVisible and 0.5 or 1
                part.CanCollide = false
                part.Massless = true
            else
                part.Size = partName == "Head" and Vector3.new(2, 1, 1) or Vector3.new(2, 2, 1)
                part.Transparency = 1
            end
        end
    end
end

local function FindMarketToolId(ToolName)
    if not ReplicatedStorage:FindFirstChild("MarketItems") then return nil end
    for _, Item in ipairs(ReplicatedStorage.MarketItems:GetChildren()) do
        if Item.Name == ToolName and Item:FindFirstChild("owner") and Item.owner.Value == LocalPlayer.Name then
            return Item:GetAttribute("SpecialId") or Item:GetAttribute("ToolId")
        end
    end
end

local function GetUniqueToolNames()
    local ToolNames = {}
    if LocalPlayer.Character then
        local CurrentTool = LocalPlayer.Character:FindFirstChildOfClass("Tool")
        if CurrentTool then ToolNames[CurrentTool.Name] = true end
    end
    local Results = {}
    for Name in pairs(ToolNames) do table.insert(Results, Name) end
    return Results
end

local function MarketProtectTool(ToolName)
    if MarketProtectedTools[ToolName] then return end
    local ToolId
    local Connection
    if ReplicatedStorage:FindFirstChild("MarketItems") then
        Connection = ReplicatedStorage.MarketItems.ChildAdded:Connect(function(Item)
            if Item.Name == ToolName and Item:FindFirstChild("owner") and Item.owner.Value == LocalPlayer.Name then
                ToolId = Item:GetAttribute("SpecialId")
            end
        end)
    end
    pcall(function() if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("Humanoid") then LocalPlayer.Character:FindFirstChild("Humanoid"):UnequipTools() end end)
    task.wait(0.15)
    pcall(function() if ReplicatedStorage:FindFirstChild("ListWeaponRemote") then ReplicatedStorage.ListWeaponRemote:FireServer(ToolName, 676767) end end)
    task.wait(0.3)
    pcall(function() if ReplicatedStorage:FindFirstChild("BackpackRemote") then ReplicatedStorage.BackpackRemote:InvokeServer("Store", ToolName) end end)
    task.wait(3)
    if not ToolId then ToolId = FindMarketToolId(ToolName) end
    if Connection then Connection:Disconnect() end
    if ToolId then MarketProtectedTools[ToolName] = ToolId end
end

local function MarketReleaseTool(ToolName, ToolId)
    if not ToolName or not ToolId then return end
    if not ReplicatedStorage:FindFirstChild("BuyItemRemote") or not ReplicatedStorage:FindFirstChild("BackpackRemote") then return end
    pcall(function() ReplicatedStorage.BuyItemRemote:FireServer(ToolName, "Remove", ToolId) end)
    task.wait(0.5)
    pcall(function() ReplicatedStorage.BackpackRemote:InvokeServer("Grab", ToolName) end)
    MarketProtectedTools[ToolName] = nil
end

local function ProtectInventoryTools()
    for _, ToolName in ipairs(GetUniqueToolNames()) do MarketProtectTool(ToolName) end
end

local function ReleaseMarketTools()
    for ToolName, ToolId in pairs(MarketProtectedTools) do MarketReleaseTool(ToolName, ToolId) end
end

local function UpdateMarketProtection(Value)
    if not KeepToolsOnDeath then return end
    if Value > 0 and Value <= MarketHealthThreshold then
        task.spawn(ProtectInventoryTools)
    elseif Value > MarketHealthThreshold then
        task.spawn(ReleaseMarketTools)
    end
end

-- ========== FLY ==========
local FlyCamera = workspace.CurrentCamera
local FlyRunService = game:GetService("RunService")
local function flyHandleSpeed()
    if not flySwimMethod then return end
    local character = LocalPlayer.Character
    if not character or not character.Parent then return end
    local humanoid = character:FindFirstChildOfClass("Humanoid")
    local hrp = character:FindFirstChild("HumanoidRootPart")
    if not humanoid or not hrp or humanoid.Health <= 0 then return end
    humanoid:ChangeState(0)
    if flyEnabled then hrp.RotVelocity = Vector3.new(0, 0, 0); return end
end
local function flyOnCharacterAdded(char)
    if flySpeedConnection then flySpeedConnection:Disconnect(); flySpeedConnection = nil end
    pcall(function() char:WaitForChild("Humanoid", 5); char:WaitForChild("HumanoidRootPart", 5) end)
    flySpeedConnection = FlyRunService.Stepped:Connect(function()
        local ok = pcall(flyHandleSpeed)
        if not ok and flySpeedConnection then flySpeedConnection:Disconnect(); flySpeedConnection = nil end
    end)
end
local function StartFly()
    if flyEnabled then return end
    flyEnabled = true; flySwimMethod = true; getgenv().SwimMethod = true; task.wait(0.2)
    local character = LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()
    if not character then return end
    local humanoid = character:FindFirstChildOfClass("Humanoid")
    local head = character:FindFirstChild("Head")
    if not humanoid or not head then return end
    humanoid.PlatformStand = true; head.Anchored = true
    if flyCFloop then flyCFloop:Disconnect() end
    flyCFloop = FlyRunService.Heartbeat:Connect(function(deltaTime)
        if not flyEnabled or not character or not character.Parent then
            if flyCFloop then flyCFloop:Disconnect() end
            return
        end
        local moveDir = humanoid.MoveDirection
        local speed = flySpeedVal * deltaTime * 100
        local camCF = FlyCamera.CFrame
        local _, yRot, _ = camCF:ToOrientation()
        local uprightRot = CFrame.Angles(0, yRot, 0)
        local flatCamCF = CFrame.new(camCF.Position) * uprightRot
        local localInput = flatCamCF:VectorToObjectSpace(moveDir * speed)
        local worldMovement = camCF:VectorToWorldSpace(localInput)
        head.CFrame = CFrame.new(head.Position + worldMovement) * uprightRot
        FlyCamera.CameraSubject = head
    end)
end
local function StopFly()
    if not flyEnabled then return end
    flyEnabled = false; flySwimMethod = false; getgenv().SwimMethod = false
    if flyCFloop then flyCFloop:Disconnect(); flyCFloop = nil end
    local character = LocalPlayer.Character
    if character then
        local humanoid = character:FindFirstChildOfClass("Humanoid")
        local head = character:FindFirstChild("Head")
        if humanoid then humanoid.PlatformStand = false; humanoid:ChangeState(Enum.HumanoidStateType.Running) end
        if head then head.Anchored = false end
    end
end

local function GetMousePosition()

    return UserInputService:GetMouseLocation()

local function GetPartCollection(player, points)
    local collection = {}
    if not player or not player.Character then return collection end
    local pointMap = {
        head = {"Head"}, body = {"UpperTorso", "LowerTorso", "Torso", "HumanoidRootPart"},
        arms = {"LeftUpperArm", "RightUpperArm", "LeftLowerArm", "RightLowerArm", "LeftHand", "RightHand"},
        legs = {"LeftUpperLeg", "RightUpperLeg", "LeftLowerLeg", "RightLowerLeg", "LeftFoot", "RightFoot"}
    }
    local added = {}
    if type(points) == "string" then points = { points } end
    for _, point in ipairs(points or {"Head"}) do
        local names = pointMap[string.lower(point)] or {}
        for _, name in ipairs(names) do
            local part = player.Character:FindFirstChild(name)
            if part and not added[part] then added[part] = true; table.insert(collection, part) end
        end
    end
    if #collection == 0 then
        local fallback = player.Character:FindFirstChild("Head")
        if fallback then table.insert(collection, fallback) end
    end
    return collection
end

local function SelectTargetPart(player, points, priority)
    local parts = GetPartCollection(player, points)
    if #parts == 0 then return nil end
    if priority == "head" then for _, part in ipairs(parts) do if part.Name == "Head" then return part end end
    elseif priority == "body" then for _, part in ipairs(parts) do local name = part.Name; if name == "HumanoidRootPart" or name == "Torso" or name == "UpperTorso" or name == "LowerTorso" then return part end end end
    return parts[math.random(1, #parts)]
end

local function ScreenDistanceToPart(part)
    if not part then return math.huge end
    local screenPoint, onScreen = Camera:WorldToViewportPoint(part.Position)
    if not onScreen then return math.huge end
    return (Vector2.new(screenPoint.X, screenPoint.Y) - GetMousePosition()).Magnitude
end

local function GetClosestPlayerToMouseSilent()
    if not Silent.Enabled then return nil end
    local bestTarget, bestDistance = nil, math.huge
    local mousePos = GetMousePosition()
    local radius = Silent.UseFieldOfView and Silent.Radius or math.huge
    for _, player in ipairs(Players:GetPlayers()) do
        if not isValidCombatPlayer(player) then continue end
        local head = player.Character:FindFirstChild("Head")
        if not head then continue end
        local screenPos, onScreen = Camera:WorldToViewportPoint(head.Position)
        if not onScreen then continue end
        local distance = (Vector2.new(screenPos.X, screenPos.Y) - mousePos).Magnitude
        if distance > radius then continue end
        if distance < bestDistance then bestDistance = distance; bestTarget = player end
    end
    return bestTarget
end

local function IsAimlockActive()
    if Aimlock.Activation == "always" then return true
    elseif Aimlock.Activation == "mouse 1" then return UserInputService:IsMouseButtonPressed(Enum.UserInputType.MouseButton1)
    elseif Aimlock.Activation == "mouse 2" then return UserInputService:IsMouseButtonPressed(Enum.UserInputType.MouseButton2)
    elseif Aimlock.Activation == "toggle" then return Aimlock.Aiming end
    return false
end

local function GetClosestPlayerToMouseAimbot()
    if not Aimlock.Enabled or not IsAimlockActive() then return nil end
    local bestTarget, bestDistance = nil, math.huge
    local mousePos = GetMousePosition()
    local radius = Aimlock.UseFieldOfView and Aimlock.Radius or 9e9
    for _, player in ipairs(Players:GetPlayers()) do
        if not isValidCombatPlayer(player) then continue end
        local targetPart = player.Character:FindFirstChild(Aimlock.TargetPart)
        if not targetPart then continue end
        local root = player.Character:FindFirstChild("HumanoidRootPart")
        if not root then continue end
        if (root.Position - Camera.CFrame.Position).Magnitude > Aimlock.MaxDistance then continue end
        if Aimlock.WallCheck and not wallCheckCharacter(player.Character) then continue end
        local screenPoint, onScreen = Camera:WorldToViewportPoint(targetPart.Position)
        if not onScreen then continue end
        local magnitude = (Vector2.new(screenPoint.X, screenPoint.Y) - mousePos).Magnitude
        if magnitude <= radius and magnitude < bestDistance then bestDistance = magnitude; bestTarget = player end
    end
    return bestTarget
end

-- ========== PROMPT INSTANT INTERACTION ==========
local function setupPrompt(prompt, enableInstant)
    if not prompt or not prompt:IsA("ProximityPrompt") then return end
    if promptConnections[prompt] then promptConnections[prompt] = nil end
    if enableInstant then prompt.HoldDuration = 0; prompt.RequiresLineOfSight = false
    else prompt.HoldDuration = 1; prompt.RequiresLineOfSight = true end
    promptConnections[prompt] = prompt
end

local function refreshPrompts(enableInstant)
    for _, prompt in ipairs(workspace:GetDescendants()) do
        if prompt:IsA("ProximityPrompt") then setupPrompt(prompt, enableInstant) end
    end
    if promptAddedConnection then promptAddedConnection:Disconnect(); promptAddedConnection = nil end
    if enableInstant then
        promptAddedConnection = workspace.DescendantAdded:Connect(function(descendant)
            if descendant:IsA("ProximityPrompt") then setupPrompt(descendant, true) end
        end)
    end
end

local function findHouseModels()
    local found = {}
    for _, v in pairs(workspace:GetDescendants()) do
        if v:IsA("Model") or v:IsA("Folder") then
            local n = v.Name:lower()
            if n:find("house") and not n:find("pent") and not n:find("ware") then
                if v:FindFirstChildWhichIsA("ProximityPrompt", true) then table.insert(found, v) end
            end
        end
    end
    table.sort(found, function(a, b) return a.Name:lower() < b.Name:lower() end)
    return found
end
local function lootHouse(houseFolder)
    if not houseFolder then return end
    for _, item in pairs(houseFolder:GetDescendants()) do
        if item:IsA("ProximityPrompt") and item.Enabled then
            local part = item.Parent
            if part and part:IsA("BasePart") then
                Config:Teleport(part.CFrame + Vector3.new(0, 3, 0)); task.wait(0.2)
                pcall(function() fireproximityprompt(item) end); task.wait(0.25)
            end
        end
    end
end
