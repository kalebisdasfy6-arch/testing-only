local function formatMoney(amount)
    local formatted = tostring(math.floor(tonumber(amount) or 0))
    while true do
        local newFormatted, count = formatted:gsub("^(-?%d+)(%d%d%d)", "%1,%2")
        formatted = newFormatted
        if count == 0 then break end
    end
    return "$" .. formatted
end

local function normalizeDropdownValue(value)
    if type(value) == "table" then
        if value[1] then return value[1] end
        for key, enabled in pairs(value) do
            if enabled then return key end
        end
        return nil
    end
    return value
end

local function setLibraryLabel(label, text)
    if not label then return end
    pcall(function()
        if type(label.Set) == "function" then label:Set(text); return end
        if type(label.SetText) == "function" then label:SetText(text); return end
        if label.Instance and label.Instance.Text ~= nil then label.Instance.Text = text; return end
        if label.Elements and label.Elements["Text"] and label.Elements["Text"].Instance then
            label.Elements["Text"].Instance.Text = text; return
        end
        if label.Text ~= nil then label.Text = text end
    end)
end

local function valueToNumber(obj)
    if not obj then return nil end
    local ok, val = pcall(function() return obj.Value end)
    if ok and tonumber(val) then return tonumber(val) end
    return nil
end

local function findMoneyValue(container, names, recursive)
    if not container then return nil end
    local lookup = {}
    for _, name in ipairs(names) do lookup[string.lower(name)] = true end

    for _, child in ipairs(container:GetChildren()) do
        if lookup[string.lower(child.Name)] then
            local val = valueToNumber(child)
            if val ~= nil then return val end
        end
    end

    if recursive then
        for _, child in ipairs(container:GetDescendants()) do
            if lookup[string.lower(child.Name)] then
                local val = valueToNumber(child)
                if val ~= nil then return val end
            end
        end
    end
    return nil
end

local function getMyMoneyValues()
    local cleanNames = {"Money", "Cash", "Wallet", "Bank", "Coins"}
    local dirtyNames = {"FilthyStack", "FilthyMoney", "DirtyMoney", "DirtyCash", "IllegalMoney", "Dirty", "Filthy"}

    local stored = LocalPlayer:FindFirstChild("stored") or LocalPlayer:FindFirstChild("Stored")
    local leaderstats = LocalPlayer:FindFirstChild("leaderstats") or LocalPlayer:FindFirstChild("Leaderstats") or LocalPlayer:FindFirstChild("stats")

    local clean = findMoneyValue(stored, {"Money"}, false)
        or findMoneyValue(leaderstats, cleanNames, false)
        or findMoneyValue(LocalPlayer, cleanNames, false)
        or findMoneyValue(LocalPlayer, cleanNames, true)

    local dirty = findMoneyValue(stored, {"FilthyStack"}, false)
        or findMoneyValue(stored, dirtyNames, false)
        or findMoneyValue(LocalPlayer, dirtyNames, false)
        or findMoneyValue(LocalPlayer, dirtyNames, true)

    return clean, dirty
end

local function MarketProtectTool(ToolName)
    if MarketProtectedTools[ToolName] then return end
    local ToolId
    local Connection
    if ReplicatedStorage:FindFirstChild("MarketItems") then
        Connection = ReplicatedStorage.MarketItems.ChildAdded:Connect(function(Item)
            if Item.Name == ToolName and Item:FindFirstChild("owner") and Item.owner.Value == LocalPlayer.Name then
                ToolId = Item:GetAttribute("SpecialId") or Item:GetAttribute("ToolId")
            end
        end)
    end
    pcall(function()
        if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("Humanoid") then
            LocalPlayer.Character.Humanoid:UnequipTools()
        end
    end)
    task.wait(0.15)
    pcall(function()
        if ReplicatedStorage:FindFirstChild("ListWeaponRemote") then
            ReplicatedStorage.ListWeaponRemote:FireServer(ToolName, 676767)
        end
    end)
    task.wait(0.3)
    pcall(function()
        if ReplicatedStorage:FindFirstChild("BackpackRemote") then
            ReplicatedStorage.BackpackRemote:InvokeServer("Store", ToolName)
        end
    end)
    task.wait(3)
    if not ToolId then ToolId = FindMarketToolId(ToolName) end
    if Connection then Connection:Disconnect() end
    if ToolId then MarketProtectedTools[ToolName] = ToolId end
end

local function MarketReleaseTool(ToolName, ToolId)
    if not ToolName or not ToolId then return end
    if not ReplicatedStorage:FindFirstChild("BuyItemRemote") or not ReplicatedStorage:FindFirstChild("BackpackRemote") then return end
    pcall(function() ReplicatedStorage.BuyItemRemote:FireServer(ToolName, "Remove", ToolId) end)
    task.wait(0.5)
    pcall(function() ReplicatedStorage.BackpackRemote:InvokeServer("Grab", ToolName) end)
    MarketProtectedTools[ToolName] = nil
end

local function ProtectInventoryTools()
    for _, ToolName in ipairs(GetUniqueToolNames()) do MarketProtectTool(ToolName) end
end

local function ReleaseMarketTools()
    for ToolName, ToolId in pairs(MarketProtectedTools) do MarketReleaseTool(ToolName, ToolId) end
end

local function UpdateMarketProtection(Value)
    if not KeepToolsOnDeath then return end
    if Value > 0 and Value <= MarketHealthThreshold then
        task.spawn(ProtectInventoryTools)
    elseif Value > MarketHealthThreshold then
        task.spawn(ReleaseMarketTools)
    end
end

local function startDumpsterAutofarm()
    if dumpsterTask then return end
    dumpsterIsRunning = true
    dumpsterTask = task.spawn(function()
        while dumpsterIsRunning do
            local dumpsters = {}
            for _, v in pairs(workspace:GetDescendants()) do
                if v:IsA("Part") and v.Name == "DumpsterPromt" then
                    local prompt = v:FindFirstChildWhichIsA("ProximityPrompt")
                    if prompt then table.insert(dumpsters, { part = v, prompt = prompt }) end
                end
            end
            for _, entry in ipairs(dumpsters) do
                if not dumpsterIsRunning then continue end
                if entry.part and entry.prompt and entry.part:IsDescendantOf(workspace) then
                    SafeTeleportTo(CFrame.new(entry.part.Position + Vector3.new(0, 3, 0)))
                    task.wait(0.6)
                    pcall(function() fireproximityprompt(entry.prompt) end)
                    task.wait(1)
                end
            end
            task.wait(2)
        end
        dumpsterTask = nil
    end)
    fireRemoteForAllBackpackItems()
end

end

local function startConstructionAutofarm()
    if constructionThread then return end

        while constructionRunning do
            task.wait(1)
            pcall(function()
                if not LocalPlayer.Character or not LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then return end
                if not LocalPlayer.Character:FindFirstChild("Humanoid") or LocalPlayer.Character:FindFirstChild("Humanoid").Health == 0 then return end

                if not LocalPlayer:GetAttribute("WorkingJob") then
                    Config:Teleport(CFrame.new(-1729, 371, -1171))
                    task.wait(0.4)
                    fireproximityprompt(workspace.ConstructionStuff["Start Job"].Prompt)
                    repeat task.wait() until LocalPlayer:GetAttribute("WorkingJob") or not constructionRunning
                end

                if not LocalPlayer.Backpack:FindFirstChild("PlyWood") and not LocalPlayer.Character:FindFirstChild("PlyWood") then
                    Config:Teleport(CFrame.new(-1728, 371, -1178))
                    repeat task.wait() fireproximityprompt(workspace.ConstructionStuff["Grab Wood"].Prompt) until LocalPlayer.Backpack:FindFirstChild("PlyWood") or LocalPlayer.Character:FindFirstChild("PlyWood") or not constructionRunning
                end

                repeat task.wait() until LocalPlayer.Backpack:FindFirstChild("PlyWood") or LocalPlayer.Character:FindFirstChild("PlyWood") or not constructionRunning

                if LocalPlayer.Backpack:FindFirstChild("PlyWood") then
                    LocalPlayer.Character.Humanoid:EquipTool(LocalPlayer.Backpack:FindFirstChild("PlyWood"))
                end

                local targetWood = nil
                for _, v in pairs(workspace.ConstructionStuff:GetDescendants()) do
                    if v:IsA("ProximityPrompt") and v.ActionText == "Wall" then
                        targetWood = v
                        break
                    end
                end

                if targetWood and targetWood.Parent then
                    Config:Teleport(targetWood.Parent.CFrame + Vector3.new(0, 3, 0))
                    task.wait(0.4)
                    fireproximityprompt(targetWood)
                    task.wait(1)
                end
            end)
        end
        constructionThread = nil
    end)
end

local function findHouseModels()
    local found = {}
    for _, v in pairs(workspace:GetDescendants()) do
        if v:IsA("Model") or v:IsA("Folder") then
            local n = v.Name:lower()
            if n:find("house") and not n:find("pent") and not n:find("ware") then
                if v:FindFirstChildWhichIsA("ProximityPrompt", true) then table.insert(found, v) end
            end
        end
    end
    table.sort(found, function(a, b) return a.Name:lower() < b.Name:lower() end)
    return found
end

local function GetPlayerMoneyAmount()
    local pl = Players.LocalPlayer
    if not pl then return nil end
    local leader = pl:FindFirstChild("leaderstats") or pl:FindFirstChild("Leaderstats") or pl:FindFirstChild("stats")
    if leader then
        for _, name in ipairs({"Money","Cash","Bank","Coins","Wallet"}) do
            local val = leader:FindFirstChild(name)
            if val and type(val.Value) == "number" then return val.Value end
        end
    end
    if pl.GetAttribute then
        local a = pl:GetAttribute("Money") or pl:GetAttribute("Cash") or pl:GetAttribute("Coins")
        if a and tonumber(a) then return tonumber(a) end
    end
    return nil
end

local WeaponMods = { HitboxSize = 1, InfiniteAmmo = false, InfiniteClips = false, Bullets80k = false, ModifyRecoilValue = false, ModifySpreadValue = false, DisableJamming = false, InstantReload = false, InstantEquip = false, ModifyFireRate = false, Automatic = false, DamageAmplified = false, SpoofAccuracy = false, HitboxExpander = false, HitboxVisible = false, Hitmarkers = false, ReloadSpeed = 0.2, EquipSpeed = 0.2 }

local function getEquippedGunSetting()
    local character = LocalPlayer.Character
    if not character then return nil, nil end
    local tool = character:FindFirstChildOfClass("Tool")
    if not tool or not tool:FindFirstChild("Setting") then return nil, nil end
    local ok, settings = pcall(require, tool.Setting)
    if ok and type(settings) == "table" then return tool, settings end
    return nil, nil
end

local function applyAllWeaponMods()
    local tool, settings = getEquippedGunSetting()
    if not settings then return end
    pcall(function()
        if WeaponMods.InfiniteAmmo then settings.Ammo = 99999; settings.AmmoPerMag = 99999; settings.LimitedAmmoEnabled = false end
        if WeaponMods.InfiniteClips then settings.Ammo = 80000; settings.AmmoPerMag = 80000; settings.LimitedAmmoEnabled = false end
        if WeaponMods.Bullets80k then
            settings.Ammo = 80000; settings.AmmoPerMag = 80000; settings.LimitedAmmoEnabled = false
            local gunLocal = tool:FindFirstChild("GunScript_Local")
            if gunLocal and type(getsenv) == "function" then
                pcall(function()
                    local env = getsenv(gunLocal)
                    if env and type(env.Reload) == "function" and type(debug.setupvalue) == "function" then
                        debug.setupvalue(env.Reload, 1, 80000); debug.setupvalue(env.Reload, 3, 80000)
                    end
                end)
            end
        end
        if WeaponMods.ModifyRecoilValue then settings.Recoil = 0; settings.CameraRecoilingEnabled = false end
        if WeaponMods.ModifySpreadValue then settings.Spread = 0; settings.SpreadX = 0; settings.SpreadY = 0; settings.Accuracy = 1 end
        if WeaponMods.DisableJamming then settings.JamChance = 0 end
        if WeaponMods.InstantReload then settings.ReloadTime = WeaponMods.ReloadSpeed or 0.1 end
        if WeaponMods.InstantEquip then settings.EquipTime = WeaponMods.EquipSpeed or 0.1; settings.EquippingTime = WeaponMods.EquipSpeed or 0.1 end
        if WeaponMods.ModifyFireRate then settings.FireRate = 0 end
        if WeaponMods.Automatic then settings.Auto = true end
        if WeaponMods.DamageAmplified then settings.BaseDamage = 10000 end
        if WeaponMods.SpoofAccuracy then settings.Accuracy = 1 end
    end)
end

MianiWeaponOriginals = MianiWeaponOriginals or {}
WeaponMods.FireRateSpeed = WeaponMods.FireRateSpeed or 50

WeaponMods.RecoilPercentage = WeaponMods.RecoilPercentage or 50

WeaponMods.KillAura = WeaponMods.KillAura or false
WeaponMods.KillAuraRange = WeaponMods.KillAuraRange or 500

function MianiGetWeaponOriginals(tool, settings)
    if not tool or not settings then return nil end
    MianiWeaponOriginals[tool.Name] = MianiWeaponOriginals[tool.Name] or {}
    for key, value in pairs(settings) do
        if MianiWeaponOriginals[tool.Name][key] == nil then
            MianiWeaponOriginals[tool.Name][key] = value
        end
    end
    return MianiWeaponOriginals[tool.Name]
end

function MianiPercent(defaultValue, percent)
    if type(defaultValue) ~= "number" then return defaultValue end
    return defaultValue * (math.clamp(tonumber(percent) or 100, 0, 100) / 100)
end

function applyBlueWeaponMods()
    local tool, settings = getEquippedGunSetting()
    if not tool or not settings then return end
    local old = MianiGetWeaponOriginals(tool, settings)
    if not old then return end

    pcall(function()
        if WeaponMods.InfiniteAmmo then
            settings.Ammo = 99999
            settings.AmmoPerMag = 99999
            settings.LimitedAmmoEnabled = false
        end
        if WeaponMods.InfiniteClips then
            settings.Ammo = 80000
            settings.AmmoPerMag = 80000
            settings.LimitedAmmoEnabled = false
        end
        if WeaponMods.Bullets80k then
            settings.Ammo = 80000
            settings.AmmoPerMag = 80000
            settings.LimitedAmmoEnabled = false
            local gunLocal = tool:FindFirstChild("GunScript_Local")
            if gunLocal and type(getsenv) == "function" then
                pcall(function()
                    local env = getsenv(gunLocal)
                    if env and type(env.Reload) == "function" and type(debug.setupvalue) == "function" then
                        debug.setupvalue(env.Reload, 1, 80000)
                        debug.setupvalue(env.Reload, 3, 80000)
                    end
                end)
            end
        end
        if WeaponMods.ModifyFireRate then
            settings.FireRate = MianiPercent(old.FireRate or settings.FireRate or 0, WeaponMods.FireRateSpeed)
        end
        if WeaponMods.InstantReload then
            settings.ReloadTime = 0.01
        elseif WeaponMods.ReloadSpeed then
            settings.ReloadTime = WeaponMods.ReloadSpeed
        end
        if WeaponMods.InstantEquip then
            settings.EquipTime = 0.01
            settings.EquippingTime = 0.01
        elseif WeaponMods.EquipSpeed then
            settings.EquipTime = WeaponMods.EquipSpeed
            settings.EquippingTime = WeaponMods.EquipSpeed
        end
        if WeaponMods.ModifySpreadValue then
            if settings.SpreadXY ~= nil then settings.SpreadXY = MianiPercent(old.SpreadXY or settings.SpreadXY, WeaponMods.SpreadPercentage) end
            if settings.SpreadYX ~= nil then settings.SpreadYX = MianiPercent(old.SpreadYX or settings.SpreadYX, WeaponMods.SpreadPercentage) end
            if settings.Spread ~= nil then settings.Spread = MianiPercent(old.Spread or settings.Spread, WeaponMods.SpreadPercentage) end
            if settings.SpreadX ~= nil then settings.SpreadX = MianiPercent(old.SpreadX or settings.SpreadX, WeaponMods.SpreadPercentage) end
            if settings.SpreadY ~= nil then settings.SpreadY = MianiPercent(old.SpreadY or settings.SpreadY, WeaponMods.SpreadPercentage) end
            settings.Accuracy = 1
        end
        if WeaponMods.ModifyRecoilValue then
            if settings.Recoil ~= nil then settings.Recoil = MianiPercent(old.Recoil or settings.Recoil, WeaponMods.RecoilPercentage) end
            settings.CameraRecoilingEnabled = false
        end
        if WeaponMods.Automatic then settings.Auto = true end
        if WeaponMods.DisableJamming then settings.JamChance = 0 end
        if WeaponMods.InfiniteDamage or WeaponMods.DamageAmplified then settings.BaseDamage = math.huge end
        if WeaponMods.SpoofAccuracy then settings.Accuracy = 1 end
    end)
end

function MianiApplyHitboxExpander()
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and player.Character then
            local part = player.Character:FindFirstChild("HumanoidRootPart") or player.Character:FindFirstChild("Head")
            if part and part:IsA("BasePart") then
                if WeaponMods.HitboxExpander then
                    local size = tonumber(WeaponMods.HitboxSize) or 5
                    part.Size = Vector3.new(size, size, size)
                    part.Transparency = WeaponMods.HitboxVisible and 0.5 or 1
                    part.Material = Enum.Material.ForceField
                    part.CanCollide = false
                    part.Massless = true
                else
                    part.Transparency = 1
                end
            end
        end
    end
end

            else
                v.Color = GunChamsColor or Color3.fromRGB(0, 200, 0)

            v.Material = Enum.Material.Neon
        end
    end
end

        end

    GunOriginalColors = {}
end

local function StopCarFly()
    CarFly.Enabled = false
    if carFlyConnection then
        carFlyConnection:Disconnect()
        carFlyConnection = nil
    end
    for _, v in pairs(workspace:GetDescendants()) do
        if v.Name == "SleepyHubCarVelocity" or v.Name == "SleepyHubCarGyro" then
            v:Destroy()
        end
    end
end

local function StartCarFly()
    local char = LocalPlayer.Character
    if not char then return end
    local humanoid = char:FindFirstChildWhichIsA("Humanoid")
    if not humanoid then return end
    local seat = humanoid.SeatPart
    if not seat or not seat:IsA("VehicleSeat") then
        sendNotification("Car Fly", "Sit in a vehicle first!")
        return
    end

    local root = (seat.Parent and seat.Parent.PrimaryPart) or seat
    StopCarFly()
    CarFly.Enabled = true

    local bv = Instance.new("BodyVelocity")
    bv.Name = "SleepyHubCarVelocity"
    bv.MaxForce = Vector3.new(1e9, 1e9, 1e9)
    bv.Parent = root

    local bg = Instance.new("BodyGyro")
    bg.Name = "SleepyHubCarGyro"
    bg.MaxTorque = Vector3.new(1e9, 1e9, 1e9)
    bg.P = 50000
    bg.D = 1000
    bg.CFrame = root.CFrame
    bg.Parent = root

    carFlyConnection = RunService.RenderStepped:Connect(function()
        if not CarFly.Enabled then StopCarFly() return end
        if not seat.Parent then StopCarFly() return end

        local cam = workspace.CurrentCamera
        local moveDir = Vector3.zero
        local look = cam.CFrame.LookVector
        local right = cam.CFrame.RightVector

        if UserInputService:IsKeyDown(Enum.KeyCode.W) then moveDir = moveDir + look end
        if UserInputService:IsKeyDown(Enum.KeyCode.S) then moveDir = moveDir - look end
        if UserInputService:IsKeyDown(Enum.KeyCode.A) then moveDir = moveDir - right end
        if UserInputService:IsKeyDown(Enum.KeyCode.D) then moveDir = moveDir + right end
        if UserInputService:IsKeyDown(Enum.KeyCode.Space) then moveDir = moveDir + Vector3.new(0, 1, 0) end
        if UserInputService:IsKeyDown(Enum.KeyCode.LeftShift) then moveDir = moveDir - Vector3.new(0, 1, 0) end

        if moveDir.Magnitude > 0 then
            bv.Velocity = moveDir.Unit * CarFly.Speed
            bg.CFrame = CFrame.new(root.Position, root.Position + moveDir)
        else
            bv.Velocity = Vector3.zero
        end
    end)
end

local function GetNearestCar(hrp)
    local CivCars = workspace:FindFirstChild("CivCars")
    if not CivCars then return nil end

    local nearestCar = nil
    local nearestDist = math.huge
    for _, car in ipairs(CivCars:GetChildren()) do
        if car:IsA("Model") then
            local seat = car:FindFirstChild("DriveSeat") or car:FindFirstChildWhichIsA("VehicleSeat")
            if seat and not seat.Occupant then
                if not car.PrimaryPart then
                    pcall(function()
                        car.PrimaryPart = car:FindFirstChildWhichIsA("BasePart")
                    end)
                end
                if car.PrimaryPart then
                    local dist = (car.PrimaryPart.Position - hrp.Position).Magnitude
                    if dist < nearestDist then
                        nearestDist = dist
                        nearestCar = car
                    end
                end
            end
        end
    end
    return nearestCar
end

local function buyShopItem(selectedShopItem)
    if not selectedShopItem then sendNotification("Shop", "Select an item first!"); return end
    local itemName = selectedShopItem
    local cleanName = string.match(itemName, "^(.*) %-")
    if cleanName then itemName = cleanName:match("^%s*(.-)%s*$") end

    local promptPurchases = workspace:FindFirstChild("PromptPurchases")
    if promptPurchases and promptPurchases:FindFirstChild(itemName) then
        local entry = promptPurchases[itemName]
        local proxprompt = entry:FindFirstChild("proxprompt")
        local prompt = proxprompt and proxprompt:FindFirstChildOfClass("ProximityPrompt")
        if prompt then
            local part = proxprompt:IsA("Part") and proxprompt.CFrame or (entry:IsA("BasePart") and entry.CFrame)
            if part then
                Config:Teleport(part)
                task.wait(0.4)
                repeat pcall(function() fireproximityprompt(prompt) end); task.wait(0.1) until LocalPlayer.Backpack:FindFirstChild(itemName) or not LocalPlayer.Character
                sendNotification("Shop", "Purchased " .. itemName)
                return
            end
        end
    end

    local gunsFolder = workspace:FindFirstChild("GUNS")
    if gunsFolder and gunsFolder:FindFirstChild(itemName) then
        local entry = gunsFolder[itemName]
        local prompt = entry:FindFirstChildWhichIsA("ProximityPrompt", true)
        if prompt then
            local part = prompt.Parent:IsA("BasePart") and prompt.Parent.CFrame or entry:GetPivot()
            Config:Teleport(part)
            task.wait(0.4)
            repeat pcall(function() fireproximityprompt(prompt) end); task.wait(0.1) until LocalPlayer.Backpack:FindFirstChild(itemName) or not LocalPlayer.Character
            sendNotification("Shop", "Purchased " .. itemName)
            return
        end
    end

    local shopRemote = ReplicatedStorage:FindFirstChild("ShopRemote")
    if shopRemote then
        local success = pcall(function() shopRemote:InvokeServer(itemName) end)
        sendNotification("Shop", success and ("Purchased " .. itemName) or ("Failed: " .. itemName))
        return
    end
    sendNotification("Shop", "Could not find purchase point for " .. itemName)
end

local function applySky(choice)
    choice = normalizeDropdownValue(choice) or tostring(choice)
    if choice == "None" then
        pcall(function()
            if originalSkySettings.ClockTime then Lighting.ClockTime = originalSkySettings.ClockTime end
            if originalSkySettings.Brightness then Lighting.Brightness = originalSkySettings.Brightness end
            if originalSkySettings.FogEnd then Lighting.FogEnd = originalSkySettings.FogEnd end
            if originalSkySettings.FogColor then Lighting.FogColor = originalSkySettings.FogColor end
            if originalSkySettings.GlobalShadows ~= nil then Lighting.GlobalShadows = originalSkySettings.GlobalShadows end
            if originalSkySettings.OutdoorAmbient then Lighting.OutdoorAmbient = originalSkySettings.OutdoorAmbient end
            if originalSkySettings.Ambient then Lighting.Ambient = originalSkySettings.Ambient end
            if originalSkySettings.ExposureCompensation then Lighting.ExposureCompensation = originalSkySettings.ExposureCompensation end
        end)
        return
    end
    local preset = skyPresets[choice]
    if not preset then return end
    pcall(function()
        Lighting.ClockTime = preset.ClockTime; Lighting.Brightness = preset.Brightness; Lighting.FogEnd = preset.FogEnd; Lighting.FogColor = preset.FogColor
        Lighting.GlobalShadows = true; Lighting.OutdoorAmbient = preset.OutdoorAmbient; Lighting.Ambient = preset.Ambient; Lighting.ExposureCompensation = preset.ExposureCompensation
        local atm = Lighting:FindFirstChildOfClass("Atmosphere") or Instance.new("Atmosphere", Lighting)
        atm.Color = preset.AtmColor; atm.Decay = preset.AtmDecay; atm.Glare = preset.AtmGlare; atm.Haze = preset.AtmHaze
    end)
end

task.spawn(function()
    while task.wait(0.5) do if currentSkyChoice ~= "None" then pcall(applySky, currentSkyChoice) end end
end)

local TargetDropdownObject = nil
local TargetPlayerNames = {}

local function getTargetPlayerList()
    local list = {}
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer then
            table.insert(list, player.Name)
        end
    end
    table.sort(list)
    return list
end

local function resolveTargetDropdownValue(value)
    if type(value) == "table" then
        if value[1] then return value[1] end
        for key, enabled in pairs(value) do
            if enabled then return key end
        end
        return nil
    end
    return value
end

local function getSelectedTargetPlayer()
    local selected = resolveTargetDropdownValue(TargetUtilities.SelectedPlayer or flags["MainTargetPlayer"])
    if typeof(selected) == "Instance" and selected:IsA("Player") then return selected end
    if type(selected) == "string" and selected ~= "" then
        return Players:FindFirstChild(selected)
    end
    return nil
end

local function refreshTargetPlayerDropdown(silent)
    TargetPlayerNames = getTargetPlayerList()
    if (not TargetUtilities.SelectedPlayer) or (not table.find(TargetPlayerNames, TargetUtilities.SelectedPlayer)) then
        TargetUtilities.SelectedPlayer = TargetPlayerNames[1]
        flags["MainTargetPlayer"] = TargetUtilities.SelectedPlayer
    end

    if TargetDropdownObject then
        pcall(function()
            if TargetDropdownObject.Refresh then TargetDropdownObject:Refresh(TargetPlayerNames) end
        end)
        pcall(function()
            if TargetDropdownObject.refresh_options then TargetDropdownObject:refresh_options(TargetPlayerNames) end
        end)
        pcall(function()
            if TargetDropdownObject.Set and TargetUtilities.SelectedPlayer then TargetDropdownObject:Set(TargetUtilities.SelectedPlayer) end
        end)
        pcall(function()
            if TargetDropdownObject.set and TargetUtilities.SelectedPlayer then TargetDropdownObject:set(TargetUtilities.SelectedPlayer) end
        end)
    end

    if not silent then
        sendNotification("Target", "Player list refreshed: " .. tostring(#TargetPlayerNames) .. " players")
    end
end


local function isValidTargetPlayer(player)
    if not player or player == LocalPlayer then return false end
    local character = player.Character
    if not character then return false end
    local humanoid = character:FindFirstChildOfClass("Humanoid")
    local root = character:FindFirstChild("HumanoidRootPart")
    return humanoid and humanoid.Health > 0 and root ~= nil
end

local function getNearestTargetPlayer(maxDistance)
    local localRoot = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
    if not localRoot then return nil end
    local nearest, nearestDistance = nil, maxDistance or math.huge
    for _, player in ipairs(Players:GetPlayers()) do
        if isValidTargetPlayer(player) then
            local distance = (player.Character.HumanoidRootPart.Position - localRoot.Position).Magnitude
            if distance < nearestDistance then
                nearest = player
                nearestDistance = distance
            end
        end
    end
    return nearest, nearestDistance
end

local function mainTargetDistanceCheck(player, distance)
    if not player or not player.Character then return false end
    local root = player.Character:FindFirstChild("HumanoidRootPart")
    local localRoot = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
    if not root or not localRoot then return false end
    return distance > (localRoot.Position - root.Position).Magnitude
end

local function getAvailableTargetVehicle()
    local civCars = workspace:FindFirstChild("CivCars")
    if not civCars then return nil end
    for _, vehicle in ipairs(civCars:GetChildren()) do
        local driveSeat = vehicle:FindFirstChild("DriveSeat") or vehicle:FindFirstChildWhichIsA("VehicleSeat")
        if driveSeat and not driveSeat.Occupant then return vehicle end
    end
    return nil
end

    if TargetUtilities.SpectatePlayer then
        targetSpectateReset = false

        if not subject and LocalPlayer.Character then subject = LocalPlayer.Character:FindFirstChildOfClass("Humanoid") end

    end
end)

                LocalPlayer.Character.Humanoid.Sit = false

            pcall(function() car.PrimaryPart = car:FindFirstChild("Body") and (car.Body:FindFirstChild("#Weight", true) or car.Body:FindFirstChildWhichIsA("BasePart", true)) or car:FindFirstChildWhichIsA("BasePart", true) end)
        end
        if car.PrimaryPart then
            car:SetPrimaryPartCFrame(target.Character.HumanoidRootPart.CFrame)
        end
    end
end)

local function createDrawing(className, props)
    if not Drawing or type(Drawing.new) ~= "function" then
        return {Visible = false, Remove = function() end}
    end
    local ok, obj = pcall(Drawing.new, className)
    if not ok or not obj then return {Visible = false, Remove = function() end} end
    for k, v in pairs(props or {}) do pcall(function() obj[k] = v end) end
    return obj
end

local function createESPObject(player)
    if player == LocalPlayer or ESPObjects[player] then return end
    local box = {}
    for _, name in ipairs({"Top", "Bottom", "Left", "Right", "TopLeft", "TopRight", "BottomLeft", "BottomRight"}) do
        box[name] = createDrawing("Line", {Visible = false, Thickness = 1, Color = ESPFlags.BoxColor})
    end
    local skeleton = {}
    for i = 1, #SkeletonPairs do
        skeleton[i] = createDrawing("Line", {Visible = false, Thickness = 1, Color = ESPFlags.SkeletonColor})
    end
    ESPObjects[player] = {
        Box = box,
        Fill = createDrawing("Square", {Visible = false, Filled = true, Transparency = 0.85, Color = ESPFlags.BoxFillColor}),
        HealthOutline = createDrawing("Square", {Visible = false, Filled = false, Color = Color3.fromRGB(0,0,0), Thickness = 1}),
        HealthFill = createDrawing("Square", {Visible = false, Filled = true, Color = Color3.fromRGB(0,255,0)}),
        HealthText = createDrawing("Text", {Visible = false, Center = true, Outline = true, Size = ESPFlags.TextSize, Color = Color3.fromRGB(255,255,255)}),
        Name = createDrawing("Text", {Visible = false, Center = true, Outline = true, Size = ESPFlags.TextSize, Color = ESPFlags.TextColor}),
        Distance = createDrawing("Text", {Visible = false, Center = true, Outline = true, Size = ESPFlags.TextSize, Color = ESPFlags.TextColor}),
        Weapon = createDrawing("Text", {Visible = false, Center = true, Outline = true, Size = ESPFlags.TextSize, Color = ESPFlags.TextColor}),
        Snapline = createDrawing("Line", {Visible = false, Thickness = 1, Color = ESPFlags.SnaplineColor}),
        LookTracer = createDrawing("Line", {Visible = false, Thickness = 1, Color = ESPFlags.LookTracerColor}),
        Skeleton = skeleton,
    }
end

local function hideESPObject(obj)
    if not obj then return end
    if obj.Box then for _, line in pairs(obj.Box) do line.Visible = false end end
    if obj.Skeleton then for _, line in ipairs(obj.Skeleton) do line.Visible = false end end
    for _, key in ipairs({"Fill", "HealthOutline", "HealthFill", "HealthText", "Name", "Distance", "Weapon", "Snapline", "LookTracer"}) do
        if obj[key] then obj[key].Visible = false end
    end
end

local function removeESPObject(player)
    local obj = ESPObjects[player]
    if obj then
        if obj.Box then for _, line in pairs(obj.Box) do pcall(function() line:Remove() end) end end
        if obj.Skeleton then for _, line in ipairs(obj.Skeleton) do pcall(function() line:Remove() end) end end
        for _, key in ipairs({"Fill", "HealthOutline", "HealthFill", "HealthText", "Name", "Distance", "Weapon", "Snapline", "LookTracer"}) do
            if obj[key] then pcall(function() obj[key]:Remove() end) end
        end
        ESPObjects[player] = nil
    end
    if HighlightObjects[player] then
        pcall(function() HighlightObjects[player]:Destroy() end)
        HighlightObjects[player] = nil
    end
end

local function getESPColor(player, defaultColor)
    if ESPFlags.TeamColor and player.Team and LocalPlayer.Team and player.Team == LocalPlayer.Team then
        return Color3.fromRGB(0,120,255)
    end
    return defaultColor
end

local function updateLine(line, from, to, color)
    if not line then return end
    line.From = from
    line.To = to
    line.Color = color
    line.Visible = true
end

RunService.RenderStepped:Connect(function()
    local camera = workspace.CurrentCamera
    if not camera then return end
    for _, player in ipairs(Players:GetPlayers()) do
        if player == LocalPlayer then continue end
        createESPObject(player)
        local obj = ESPObjects[player]
        local character = player.Character
        local humanoid = character and character:FindFirstChildOfClass("Humanoid")
        local root = character and character:FindFirstChild("HumanoidRootPart")
        if not (ESPFlags.Enabled and obj and character and humanoid and root and humanoid.Health > 0) then
            hideESPObject(obj)
            if HighlightObjects[player] then HighlightObjects[player].Enabled = false end
            continue
        end
        local localRoot = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
        local distance = localRoot and (root.Position - localRoot.Position).Magnitude or (root.Position - camera.CFrame.Position).Magnitude
        if distance > ESPFlags.RenderDistance then
            hideESPObject(obj)
            if HighlightObjects[player] then HighlightObjects[player].Enabled = false end
            continue
        end

        local rootPos, onScreen = camera:WorldToViewportPoint(root.Position)
        if not onScreen then hideESPObject(obj); continue end

        local size = character:GetExtentsSize()
        local top = camera:WorldToViewportPoint((root.CFrame * CFrame.new(0, size.Y / 2, 0)).Position)
        local bottom = camera:WorldToViewportPoint((root.CFrame * CFrame.new(0, -size.Y / 2, 0)).Position)
        local height = math.max(math.abs(bottom.Y - top.Y), 2)
        local width = math.max(height * 0.58, 2)
        local x = top.X - width / 2
        local y = top.Y
        local boxColor = getESPColor(player, ESPFlags.BoxColor)
        local textColor = ESPFlags.TextColor

        if ESPFlags.BoxFill then
            obj.Fill.Position = Vector2.new(x, y)
            obj.Fill.Size = Vector2.new(width, height)
            obj.Fill.Color = ESPFlags.BoxFillColor
            obj.Fill.Transparency = 0.85
            obj.Fill.Visible = true
        else
            obj.Fill.Visible = false
        end

        if ESPFlags.Boxes then
            local tl, tr = Vector2.new(x, y), Vector2.new(x + width, y)
            local bl, br = Vector2.new(x, y + height), Vector2.new(x + width, y + height)
            local corner = math.min(width, height) * 0.28
            if ESPFlags.BoxType == "Full" then
                updateLine(obj.Box.Top, tl, tr, boxColor); updateLine(obj.Box.Bottom, bl, br, boxColor)
                updateLine(obj.Box.Left, tl, bl, boxColor); updateLine(obj.Box.Right, tr, br, boxColor)
                obj.Box.TopLeft.Visible = false; obj.Box.TopRight.Visible = false; obj.Box.BottomLeft.Visible = false; obj.Box.BottomRight.Visible = false
            else
                obj.Box.Top.Visible = false; obj.Box.Bottom.Visible = false; obj.Box.Left.Visible = false; obj.Box.Right.Visible = false
                updateLine(obj.Box.TopLeft, tl, tl + Vector2.new(corner, 0), boxColor)
                updateLine(obj.Box.Left, tl, tl + Vector2.new(0, corner), boxColor)
                updateLine(obj.Box.TopRight, tr, tr - Vector2.new(corner, 0), boxColor)
                updateLine(obj.Box.Right, tr, tr + Vector2.new(0, corner), boxColor)
                updateLine(obj.Box.BottomLeft, bl, bl + Vector2.new(corner, 0), boxColor)
                updateLine(obj.Box.Bottom, bl, bl - Vector2.new(0, corner), boxColor)
                updateLine(obj.Box.BottomRight, br, br - Vector2.new(corner, 0), boxColor)
                updateLine(obj.Box.Top, br, br - Vector2.new(0, corner), boxColor)
            end
        else
            for _, line in pairs(obj.Box) do line.Visible = false end
        end

        if ESPFlags.Healthbar then
            local pct = math.clamp(humanoid.Health / math.max(humanoid.MaxHealth, 1), 0, 1)
            local barHeight = height * pct
            obj.HealthOutline.Position = Vector2.new(x - 7, y)
            obj.HealthOutline.Size = Vector2.new(ESPFlags.HealthbarThickness + 2, height)
            obj.HealthOutline.Visible = true
            obj.HealthFill.Position = Vector2.new(x - 6, y + height - barHeight)
            obj.HealthFill.Size = Vector2.new(ESPFlags.HealthbarThickness, barHeight)
            obj.HealthFill.Color = Color3.fromRGB(255 - (255 * pct), 255 * pct, 0)
            obj.HealthFill.Visible = true
            if ESPFlags.HealthbarNumber then
                obj.HealthText.Text = tostring(math.floor(humanoid.Health))
                obj.HealthText.Position = Vector2.new(x - 12, y + height - barHeight)
                obj.HealthText.Size = ESPFlags.TextSize
                obj.HealthText.Visible = true
            else
                obj.HealthText.Visible = false
            end
        else
            obj.HealthOutline.Visible = false; obj.HealthFill.Visible = false; obj.HealthText.Visible = false
        end

        local bottomOffset = 2
        if ESPFlags.Name then
            obj.Name.Text = player.DisplayName ~= player.Name and (player.DisplayName .. " (@" .. player.Name .. ")") or player.Name
            obj.Name.Position = Vector2.new(x + width / 2, y - ESPFlags.TextSize - 2)
            obj.Name.Size = ESPFlags.TextSize
            obj.Name.Color = textColor
            obj.Name.Visible = true
        else obj.Name.Visible = false end

        if ESPFlags.Distance then
            obj.Distance.Text = tostring(math.floor(distance)) .. "m"
            obj.Distance.Position = Vector2.new(x + width / 2, y + height + bottomOffset)
            obj.Distance.Size = ESPFlags.TextSize
            obj.Distance.Color = textColor
            obj.Distance.Visible = true
            bottomOffset += ESPFlags.TextSize + 1
        else obj.Distance.Visible = false end

        if ESPFlags.Weapon then
            local tool = character:FindFirstChildOfClass("Tool")
            obj.Weapon.Text = tool and ("[" .. tool.Name .. "]") or "[None]"
            obj.Weapon.Position = Vector2.new(x + width / 2, y + height + bottomOffset)
            obj.Weapon.Size = ESPFlags.TextSize
            obj.Weapon.Color = textColor
            obj.Weapon.Visible = true
        else obj.Weapon.Visible = false end

        if ESPFlags.Snaplines then
            obj.Snapline.From = Vector2.new(camera.ViewportSize.X / 2, camera.ViewportSize.Y)
            obj.Snapline.To = Vector2.new(rootPos.X, rootPos.Y)
            obj.Snapline.Color = getESPColor(player, ESPFlags.SnaplineColor)
            obj.Snapline.Visible = true
        else obj.Snapline.Visible = false end

        if ESPFlags.LookTracers and character:FindFirstChild("Head") then
            local head = character.Head
            local p1, o1 = camera:WorldToViewportPoint(head.Position)
            local p2, o2 = camera:WorldToViewportPoint(head.Position + head.CFrame.LookVector * 10)
            if o1 and o2 then
                obj.LookTracer.From = Vector2.new(p1.X, p1.Y)
                obj.LookTracer.To = Vector2.new(p2.X, p2.Y)
                obj.LookTracer.Color = ESPFlags.LookTracerColor
                obj.LookTracer.Visible = true
            else obj.LookTracer.Visible = false end
        else obj.LookTracer.Visible = false end

        if ESPFlags.Skeleton then
            for i, pair in ipairs(SkeletonPairs) do
                local a, b = character:FindFirstChild(pair[1]), character:FindFirstChild(pair[2])
                if a and b and obj.Skeleton[i] then
                    local pa, oa = camera:WorldToViewportPoint(a.Position)
                    local pb, ob = camera:WorldToViewportPoint(b.Position)
                    if oa and ob then
                        obj.Skeleton[i].From = Vector2.new(pa.X, pa.Y)
                        obj.Skeleton[i].To = Vector2.new(pb.X, pb.Y)
                        obj.Skeleton[i].Color = ESPFlags.SkeletonColor
                        obj.Skeleton[i].Visible = true
                    else obj.Skeleton[i].Visible = false end
                elseif obj.Skeleton[i] then obj.Skeleton[i].Visible = false end
            end
        else
            for _, line in ipairs(obj.Skeleton) do line.Visible = false end
        end

        if ESPFlags.Chams then
            local highlight = HighlightObjects[player]
            if not highlight or not highlight.Parent then
                highlight = Instance.new("Highlight")
                highlight.Name = "MianiBlueESPHighlight"
                highlight.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
                highlight.Parent = CoreGui
                HighlightObjects[player] = highlight
            end
            highlight.Adornee = character
            highlight.FillColor = getESPColor(player, ESPFlags.ChamsColor)
            highlight.OutlineColor = Color3.fromRGB(255,255,255)
            highlight.FillTransparency = 0.55
            highlight.OutlineTransparency = 0.1
            highlight.Enabled = true
        elseif HighlightObjects[player] then
            HighlightObjects[player].Enabled = false
        end
    end
end)

local function drawBulletTracer(endPos)
    if not VisualTracers.Enabled then return end
    local character = LocalPlayer.Character
    local tool = character and character:FindFirstChildOfClass("Tool")
    local handle = tool and tool:FindFirstChild("Handle")
    if not handle then return end
    local muzzle = handle:FindFirstChild("GunMuzzlePoint") or handle:FindFirstChild("GunFirePoint")
    local startPos = muzzle and muzzle:IsA("Attachment") and muzzle.WorldPosition or handle.Position
    local startPart = Instance.new("Part")
    startPart.Anchored = true; startPart.Transparency = 1; startPart.CanCollide = false; startPart.CanQuery = false; startPart.CanTouch = false
    startPart.Position = startPos; startPart.Parent = workspace
    local endPart = Instance.new("Part")
    endPart.Anchored = true; endPart.Transparency = 1; endPart.CanCollide = false; endPart.CanQuery = false; endPart.CanTouch = false
    endPart.Position = endPos; endPart.Parent = workspace
    local a0 = Instance.new("Attachment", startPart)
    local a1 = Instance.new("Attachment", endPart)
    local beam = Instance.new("Beam", startPart)
    beam.Attachment0 = a0; beam.Attachment1 = a1; beam.FaceCamera = true; beam.LightEmission = 1; beam.Width0 = 0.08; beam.Width1 = 0.04
    if VisualTracers.Rainbow then
        task.spawn(function()
            while beam and beam.Parent do
                local h = (tick() % 5) / 5
                beam.Color = ColorSequence.new(Color3.fromHSV(h,1,1), Color3.fromHSV((h + 0.12) % 1,1,1))
                task.wait(0.05)
            end
        end)
    else
        beam.Color = ColorSequence.new(VisualTracers.StartColor, VisualTracers.EndColor)
    end
    task.delay(VisualTracers.Duration, function()
        pcall(function() startPart:Destroy() end)
        pcall(function() endPart:Destroy() end)
    end)
end
